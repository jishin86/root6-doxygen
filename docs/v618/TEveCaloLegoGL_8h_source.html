145 inline void TEveCaloLegoGL::WrapTwoPi(Float_t &min, Float_t &max) const
146 {
147  if (fM->GetData()->GetWrapTwoPi())
148  {
149  if (fM->GetPhiMax()>TMath::Pi() && max<=fM->GetPhiMin())
150  {
151  min += TMath::TwoPi();
152  max += TMath::TwoPi();
153  }
154  else if (fM->GetPhiMin()<-TMath::Pi() && min>=fM->GetPhiMax())
155  {
156  min -= TMath::TwoPi();
157  max -= TMath::TwoPi();
158  }
159  }
160 }
