29 Quaternion Quaternion::operator * (const RotationX & rx) const {
30  // combination with a RotationX
31  Quaternion q(rx);
32  return Quaternion (
33  U()*q.U() - I()*q.I()
34  , I()*q.U() + U()*q.I()

38 }
