77 static void CallEndOfProcessCleanups()
78 {
79  // Insure that the files, canvases and sockets are closed.
80 
81  // If we get here, the tear down has started. We have no way to know what
82  // has or has not yet been done. In particular on Ubuntu, this was called

85  // autoloading libraries (!) that is pointless ...
86  if (gROOT) {
87  gROOT->SetBit(kInvalidObject);
88  gROOT->EndOfProcessCleanups();

104 ////////////////////////////////////////////////////////////////////////////////
105 /// Create an application environment. The application environment
106 /// provides an interface to the graphics system and eventloop
107 /// (be it X, Windows, MacOS or BeOS). After creating the application
108 /// object start the eventloop by calling its Run() method. The command
109 /// line options recognized by TApplication are described in the GetOptions()
110 /// method. The recognized options are removed from the argument array.
111 /// The original list of argument options can be retrieved via the Argc()
112 /// and Argv() methods. The appClassName "proofserv" is reserved for the
113 /// PROOF system. The "options" and "numOptions" arguments are not used,
114 /// except if you want to by-pass the argv processing by GetOptions()
115 /// in which case you should specify numOptions<0. All options will
116 /// still be available via the Argv() method for later use.
117 
118 TApplication::TApplication(const char *appClassName, Int_t *argc, char **argv,
119  void * /*options*/, Int_t numOptions) :
120  fArgc(0), fArgv(0), fAppImp(0), fIsRunning(kFALSE), fReturnFromRun(kFALSE),
121  fNoLog(kFALSE), fNoLogo(kFALSE), fQuit(kFALSE), fUseMemstat(kFALSE),
122  fFiles(0), fIdleTimer(0), fSigHandler(0), fExitOnException(kDontExit),
123  fAppRemote(0)
124 {
125  R__LOCKGUARD(gInterpreterMutex);
126 
127  // Create the list of applications the first time
128  if (!fgApplications)
129  fgApplications = new TList;
130 
131  // Add the new TApplication early, so that the destructor of the
132  // default TApplication (if it is called in the block of code below)
133  // will not destroy the files, socket or TColor that have already been
134  // created.
135  fgApplications->Add(this);
136 
137  if (gApplication && gApplication->TestBit(kDefaultApplication)) {
138  // allow default TApplication to be replaced by a "real" TApplication
139  delete gApplication;
140  gApplication = 0;
141  gROOT->SetBatch(kFALSE);
142  fgGraphInit = kFALSE;
143  }
144 
145  if (gApplication) {
146  Error("TApplication", "only one instance of TApplication allowed");
147  fgApplications->Remove(this);
148  return;
149  }
150 
151  if (!gROOT)
152  ::Fatal("TApplication::TApplication", "ROOT system not initialized");
153 
154  if (!gSystem)
155  ::Fatal("TApplication::TApplication", "gSystem not initialized");
156 
157  static Bool_t hasRegisterAtExit(kFALSE);
158  if (!hasRegisterAtExit) {
159  // If we are the first TApplication register the atexit)
160  atexit(CallEndOfProcessCleanups);
161  hasRegisterAtExit = kTRUE;
162  }
163  gROOT->SetName(appClassName);
164 
165  // copy command line arguments, can be later accessed via Argc() and Argv()
166  if (argc && *argc > 0) {
167  fArgc = *argc;
168  fArgv = (char **)new char*[fArgc];
169  }
170 
171  for (int i = 0; i < fArgc; i++)
172  fArgv[i] = StrDup(argv[i]);
173 
174  if (numOptions >= 0)
175  GetOptions(argc, argv);
176 
177  if (fArgv)
178  gSystem->SetProgname(fArgv[0]);
179 
180  // Tell TSystem the TApplication has been created
181  gSystem->NotifyApplicationCreated();
182 
183  fAppImp = gGuiFactory->CreateApplicationImp(appClassName, argc, argv);
184  ResetBit(kProcessRemotely);
185 
186  // Initialize the graphics environment
187  if (gClassTable->GetDict("TPad")) {
188  fgGraphNeeded = kTRUE;
189  InitializeGraphics();
190  }
191 
192  // Save current interpreter context
193  gInterpreter->SaveContext();
194  gInterpreter->SaveGlobalsContext();
195 
196  // to allow user to interact with TCanvas's under WIN32
197  gROOT->SetLineHasBeenProcessed();
198 
199  // activate TMemStat
200  if (fUseMemstat || gEnv->GetValue("Root.TMemStat", 0)) {
201  fUseMemstat = kTRUE;
202  Int_t buffersize = gEnv->GetValue("Root.TMemStat.buffersize", 100000);
203  Int_t maxcalls = gEnv->GetValue("Root.TMemStat.maxcalls", 5000000);
204  const char *ssystem = gEnv->GetValue("Root.TMemStat.system","gnubuiltin");
205  if (maxcalls > 0) {
206  gROOT->ProcessLine(Form("new TMemStat(\"%s\",%d,%d);",ssystem,buffersize,maxcalls));
207  }
208  }
209 
210  //Needs to be done last
211  gApplication = this;
212  gROOT->SetApplication(this);
213 
214 }

221  for (int i = 0; i < fArgc; i++)
222  if (fArgv[i]) delete [] fArgv[i];

226  fgApplications->Remove(this);

232  }

235  // end of 'main' (or more exactly before the library start being
236  // unloaded).

239  gROOT->EndOfProcessCleanups();

242  }
243  }
244

246  // delete the implementation.
247  SafeDelete(fAppImp);

353 ////////////////////////////////////////////////////////////////////////////////
354 /// Get and handle command line options. Arguments handled are removed
355 /// from the argument array. See CommandLineOptionsHelp.h for options.
356 
357 void TApplication::GetOptions(Int_t *argc, char **argv)
358 {
359  static char null[1] = { "" };
360 
361  fNoLog = kFALSE;
362  fQuit = kFALSE;
363  fFiles = 0;
364 
365  if (!argc)
366  return;
367 
368  int i, j;
369  TString pwd;
370 
371  for (i = 1; i < *argc; i++) {
372  if (!strcmp(argv[i], "-?") || !strncmp(argv[i], "-h", 2) ||
373  !strncmp(argv[i], "--help", 6)) {
374  fprintf(stderr, kCommandLineOptionsHelp);
375  Terminate(0);
376  } else if (!strncmp(argv[i], "--version", 9)) {
377  fprintf(stderr, "ROOT Version: %s\n", gROOT->GetVersion());
378  fprintf(stderr, "Built for %s on %s\n",
379  gSystem->GetBuildArch(),
380  gROOT->GetGitDate());
381 
382  fprintf(stderr, "From %s@%s\n",
383  gROOT->GetGitBranch(),
384  gROOT->GetGitCommit());
385 
386  Terminate(0);
387  } else if (!strcmp(argv[i], "-config")) {
388  fprintf(stderr, "ROOT ./configure options:\n%s\n", gROOT->GetConfigOptions());
389  Terminate(0);
390  } else if (!strcmp(argv[i], "-memstat")) {
391  fUseMemstat = kTRUE;
392  argv[i] = null;
393  } else if (!strcmp(argv[i], "-b")) {
394  MakeBatch();
395  argv[i] = null;
396  } else if (!strcmp(argv[i], "-n")) {
397  fNoLog = kTRUE;
398  argv[i] = null;
399  } else if (!strcmp(argv[i], "-t")) {
400  ROOT::EnableImplicitMT();
401  // EnableImplicitMT() only enables thread safety if IMT was configured;
402  // enable thread safety even with IMT off:
403  ROOT::EnableThreadSafety();
404  argv[i] = null;
405  } else if (!strcmp(argv[i], "-q")) {
406  fQuit = kTRUE;
407  argv[i] = null;
408  } else if (!strcmp(argv[i], "-l")) {
409  // used by front-end program to not display splash screen
410  fNoLogo = kTRUE;
411  argv[i] = null;
412  } else if (!strcmp(argv[i], "-x")) {
413  fExitOnException = kExit;
414  argv[i] = null;
415  } else if (!strcmp(argv[i], "-splash")) {
416  // used when started by front-end program to signal that
417  // splash screen can be popped down (TRint::PrintLogo())
418  argv[i] = null;
419  } else if (strncmp(argv[i], "--web", 5) == 0) {
420  // the web mode is requested
421  const char *opt = argv[i] + 5;
422  argv[i] = null;
423  TString argw;
424  if (gROOT->IsBatch()) argw = "batch";
425  if (*opt == '=') argw.Append(opt+1);
426  if (gSystem->Load("libROOTWebDisplay") >= 0) {
427  gROOT->SetWebDisplay(argw.Data());
428  gEnv->SetValue("Gui.Factory", "web");
429  } else {
430  Error("GetOptions", "--web option not supported, ROOT should be built with at least c++14 enabled");
431  }
432  } else if (!strcmp(argv[i], "-e")) {
433  argv[i] = null;
434  ++i;
435 
436  if ( i < *argc ) {
437  if (!fFiles) fFiles = new TObjArray;
438  TObjString *expr = new TObjString(argv[i]);
439  expr->SetBit(kExpression);
440  fFiles->Add(expr);
441  argv[i] = null;
442  } else {
443  Warning("GetOptions", "-e must be followed by an expression.");
444  }
445  } else if (!strcmp(argv[i], "--")) {
446  TObjString* macro = nullptr;
447  bool warnShown = false;
448 
449  if (fFiles) {
450  for (auto f: *fFiles) {
451  TObjString* file = dynamic_cast<TObjString*>(f);
452  if (!file) {
453  if (!dynamic_cast<TNamed*>(f)) {
454  Error("GetOptions()", "Inconsistent file entry (not a TObjString)!");
455  f->Dump();
456  } // else we did not find the file.
457  continue;
458  }
459 
460  if (file->TestBit(kExpression))
461  continue;
462  if (file->String().EndsWith(".root"))
463  continue;
464  if (file->String().Contains('('))
465  continue;
466 
467  if (macro && !warnShown && (warnShown = true))
468  Warning("GetOptions", "-- is used with several macros. "
469  "The arguments will be passed to the last one.");
470 
471  macro = file;
472  }
473  }
474 
475  if (macro) {
476  argv[i] = null;
477  ++i;
478  TString& str = macro->String();
479 
480  str += '(';
481  for (; i < *argc; i++) {
482  str += argv[i];
483  str += ',';
484  argv[i] = null;
485  }
486  str.EndsWith(",") ? str[str.Length() - 1] = ')' : str += ')';
487  } else {
488  Warning("GetOptions", "no macro to pass arguments to was provided. "
489  "Everything after the -- will be ignored.");
490  for (; i < *argc; i++)
491  argv[i] = null;
492  }
493  } else if (argv[i][0] != '-' && argv[i][0] != '+') {
494  Long64_t size;
495  Long_t id, flags, modtime;
496  char *arg = strchr(argv[i], '(');
497  if (arg) *arg = '\0';
498  char *dir = gSystem->ExpandPathName(argv[i]);
499  // ROOT-9959: we do not continue if we could not expand the path
500  if (!dir) continue;
501  TUrl udir(dir, kTRUE);
502  // remove options and anchor to check the path
503  TString sfx = udir.GetFileAndOptions();
504  TString fln = udir.GetFile();
505  sfx.Replace(sfx.Index(fln), fln.Length(), "");
506  TString path = udir.GetFile();
507  if (strcmp(udir.GetProtocol(), "file")) {
508  path = udir.GetUrl();
509  path.Replace(path.Index(sfx), sfx.Length(), "");
510  }
511  // 'path' is the full URL without suffices (options and/or anchor)
512  if (arg) *arg = '(';
513  if (!arg && !gSystem->GetPathInfo(path.Data(), &id, &size, &flags, &modtime)) {
514  if ((flags & 2)) {
515  // if directory set it in fWorkDir
516  if (pwd == "") {
517  pwd = gSystem->WorkingDirectory();
518  fWorkDir = dir;
519  gSystem->ChangeDirectory(dir);
520  argv[i] = null;
521  } else if (!strcmp(gROOT->GetName(), "Rint")) {
522  Warning("GetOptions", "only one directory argument can be specified (%s)", dir);
523  }
524  } else if (size > 0) {
525  // if file add to list of files to be processed
526  if (!fFiles) fFiles = new TObjArray;
527  fFiles->Add(new TObjString(path.Data()));
528  argv[i] = null;
529  } else {
530  Warning("GetOptions", "file %s has size 0, skipping", dir);
531  }
532  } else {
533  if (TString(udir.GetFile()).EndsWith(".root")) {
534  if (!strcmp(udir.GetProtocol(), "file")) {
535  // file ending on .root but does not exist, likely a typo
536  // warn user if plain root...
537  if (!strcmp(gROOT->GetName(), "Rint"))
538  Warning("GetOptions", "file %s not found", dir);
539  } else {
540  // remote file, give it the benefit of the doubt and add it to list of files
541  if (!fFiles) fFiles = new TObjArray;
542  fFiles->Add(new TObjString(argv[i]));
543  argv[i] = null;
544  }
545  } else {
546  TString mode,fargs,io;
547  TString fname = gSystem->SplitAclicMode(dir,mode,fargs,io);
548  char *mac;
549  if (!fFiles) fFiles = new TObjArray;
550  if ((mac = gSystem->Which(TROOT::GetMacroPath(), fname,
551  kReadPermission))) {
552  // if file add to list of files to be processed
553  fFiles->Add(new TObjString(argv[i]));
554  argv[i] = null;
555  delete [] mac;
556  } else {
557  // if file add an invalid entry to list of files to be processed
558  fFiles->Add(new TNamed("NOT FOUND!", argv[i]));
559  // only warn if we're plain root,
560  // other progs might have their own params
561  if (!strcmp(gROOT->GetName(), "Rint"))
562  Warning("GetOptions", "macro %s not found", fname.Data());
563  }
564  }
565  }
566  delete [] dir;
567  }
568  // ignore unknown options
569  }
570 
571  // go back to startup directory
572  if (pwd != "")
573  gSystem->ChangeDirectory(pwd);
574 
575  // remove handled arguments from argument array
576  j = 0;
577  for (i = 0; i < *argc; i++) {
578  if (strcmp(argv[i], "")) {
579  argv[j] = argv[i];
580  j++;
581  }
582  }
583 
584  *argc = j;
585 }

619 }

725 ////////////////////////////////////////////////////////////////////////////////
726 /// Parse the content of a line starting with ".R" (already stripped-off)
727 /// The format is
728 /// ~~~ {.cpp}
729 /// [user@]host[:dir] [-l user] [-d dbg] [script]
730 /// ~~~
731 /// The variable 'dir' is the remote directory to be used as working dir.
732 /// The username can be specified in two ways, "-l" having the priority
733 /// (as in ssh).
734 /// A 'dbg' value > 0 gives increasing verbosity.
735 /// The last argument 'script' allows to specify an alternative script to
736 /// be executed remotely to startup the session.
737 
738 Int_t TApplication::ParseRemoteLine(const char *ln,
739  TString &hostdir, TString &user,
740  Int_t &dbg, TString &script)
741 {
742  if (!ln || strlen(ln) <= 0)
743  return 0;
744 
745  Int_t rc = 0;
746  Bool_t isHostDir = kTRUE;
747  Bool_t isScript = kFALSE;
748  Bool_t isUser = kFALSE;
749  Bool_t isDbg = kFALSE;
750 
751  TString line(ln);
752  TString tkn;
753  Int_t from = 0;
754  while (line.Tokenize(tkn, from, " ")) {
755  if (tkn == "-l") {
756  // Next is a user name
757  isUser = kTRUE;
758  } else if (tkn == "-d") {
759  isDbg = kTRUE;
760  } else if (tkn == "-close") {
761  rc = 1;
762  } else if (tkn.BeginsWith("-")) {
763  ::Warning("TApplication::ParseRemoteLine","unknown option: %s", tkn.Data());
764  } else {
765  if (isUser) {
766  user = tkn;
767  isUser = kFALSE;
768  } else if (isDbg) {
769  dbg = tkn.Atoi();
770  isDbg = kFALSE;
771  } else if (isHostDir) {
772  hostdir = tkn;
773  hostdir.ReplaceAll(":","/");
774  isHostDir = kFALSE;
775  isScript = kTRUE;
776  } else if (isScript) {
777  // Add everything left
778  script = tkn;
779  script.Insert(0, "\"");
780  script += "\"";
781  isScript = kFALSE;
782  break;
783  }
784  }
785  }
786 
787  // Done
788  return rc;
789 }

1037 ////////////////////////////////////////////////////////////////////////////////
1038 /// Execute a file containing a C++ macro (static method). Can be used
1039 /// while TApplication is not yet created.
1040 
1041 Long_t TApplication::ExecuteFile(const char *file, Int_t *error, Bool_t keep)
1042 {
1043  static const Int_t kBufSize = 1024;
1044 
1045  if (!file || !*file) return 0;
1046 
1047  TString aclicMode;
1048  TString arguments;
1049  TString io;
1050  TString fname = gSystem->SplitAclicMode(file, aclicMode, arguments, io);
1051 
1052  char *exnam = gSystem->Which(TROOT::GetMacroPath(), fname, kReadPermission);
1053  if (!exnam) {
1054  ::Error("TApplication::ExecuteFile", "macro %s not found in path %s", fname.Data(),
1055  TROOT::GetMacroPath());
1056  delete [] exnam;
1057  if (error)
1058  *error = (Int_t)TInterpreter::kRecoverable;
1059  return 0;
1060  }
1061 
1062  ::std::ifstream macro(exnam, std::ios::in);
1063  if (!macro.good()) {
1064  ::Error("TApplication::ExecuteFile", "%s no such file", exnam);
1065  if (error)
1066  *error = (Int_t)TInterpreter::kRecoverable;
1067  delete [] exnam;
1068  return 0;
1069  }
1070 
1071  char currentline[kBufSize];
1072  char dummyline[kBufSize];
1073  int tempfile = 0;
1074  int comment = 0;
1075  int ifndefc = 0;
1076  int ifdef = 0;
1077  char *s = 0;
1078  Bool_t execute = kFALSE;
1079  Long_t retval = 0;
1080 
1081  while (1) {
1082  bool res = (bool)macro.getline(currentline, kBufSize);
1083  if (macro.eof()) break;
1084  if (!res) {
1085  // Probably only read kBufSize, let's ignore the remainder of
1086  // the line.
1087  macro.clear();
1088  while (!macro.getline(dummyline, kBufSize) && !macro.eof()) {
1089  macro.clear();
1090  }
1091  }
1092  s = currentline;
1093  while (s && (*s == ' ' || *s == '\t')) s++; // strip-off leading blanks
1094 
1095  // very simple minded pre-processor parsing, only works in case macro file
1096  // starts with "#ifndef __CINT__". In that case everything till next
1097  // "#else" or "#endif" will be skipped.
1098  if (*s == '#') {
1099  char *cs = Compress(currentline);
1100  if (strstr(cs, "#ifndef__CINT__") ||
1101  strstr(cs, "#if!defined(__CINT__)"))
1102  ifndefc = 1;
1103  else if (ifndefc && (strstr(cs, "#ifdef") || strstr(cs, "#ifndef") ||
1104  strstr(cs, "#ifdefined") || strstr(cs, "#if!defined")))
1105  ifdef++;
1106  else if (ifndefc && strstr(cs, "#endif")) {
1107  if (ifdef)
1108  ifdef--;
1109  else
1110  ifndefc = 0;
1111  } else if (ifndefc && !ifdef && strstr(cs, "#else"))
1112  ifndefc = 0;
1113  delete [] cs;
1114  }
1115  if (!*s || *s == '#' || ifndefc || !strncmp(s, "//", 2)) continue;
1116 
1117  if (!comment && (!strncmp(s, ".X", 2) || !strncmp(s, ".x", 2))) {
1118  retval = ExecuteFile(s+3);
1119  execute = kTRUE;
1120  continue;
1121  }
1122 
1123  if (!strncmp(s, "/*", 2)) comment = 1;
1124  if (comment) {
1125  // handle slightly more complex cases like: /* */ /*
1126 again:
1127  s = strstr(s, "*/");
1128  if (s) {
1129  comment = 0;
1130  s += 2;
1131 
1132  while (s && (*s == ' ' || *s == '\t')) s++; // strip-off leading blanks
1133  if (!*s) continue;
1134  if (!strncmp(s, "//", 2)) continue;
1135  if (!strncmp(s, "/*", 2)) {
1136  comment = 1;
1137  goto again;
1138  }
1139  }
1140  }
1141  if (!comment && *s == '{') tempfile = 1;
1142  if (!comment) break;
1143  }
1144  macro.close();
1145 
1146  if (!execute) {
1147  TString exname = exnam;
1148  if (!tempfile) {
1149  // We have a script that does NOT contain an unnamed macro,
1150  // so we can call the script compiler on it.
1151  exname += aclicMode;
1152  }
1153  exname += arguments;
1154  exname += io;
1155 
1156  TString tempbuf;
1157  if (tempfile) {
1158  tempbuf.Form(".x %s", exname.Data());
1159  } else {
1160  tempbuf.Form(".X%s %s", keep ? "k" : " ", exname.Data());
1161  }
1162  retval = gInterpreter->ProcessLineSynch(tempbuf,(TInterpreter::EErrorCode*)error);
1163  }
1164 
1165  delete [] exnam;
1166  return retval;
1167 }
