42 void TGRectangle::Merge(const TGRectangle &r)
43 {
44  // Merge parameters
45  Int_t max_x = TMath::Max(fX + (Int_t) fW, r.fX + (Int_t) r.fW);
46  fX = TMath::Min(fX, r.fX);
47  Int_t max_y = TMath::Max(fY + (Int_t) fH, r.fY + (Int_t) r.fH);
48  fY = TMath::Min(fY, r.fY);
49  fW = max_x - fX;
50  fH = max_y - fY;
51 }
