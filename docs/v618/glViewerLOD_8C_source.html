144  rot = new TGeoRotation("rot",0.0, 0.0, 0.0);
145  }
146  top->AddNode(volumes[useVolume], i, new TGeoCombiTrans(*trans, *rot));
147  //printf("Added node %d (Volume %d)\n", i, useVolume);
148  }
149  geom->CloseGeometry();
