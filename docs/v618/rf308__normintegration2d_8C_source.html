43  cout << "gxy = " << gxy.getVal() << endl;
44 
45  // Return value of gxy normalized over x _and_ y in range [-10,10]
46  RooArgSet nset_xy(x, y);

48

52  cout << "gx_Int[x,y] = " << igxy->getVal() << endl;

65  // ----------------------------------------------------------------------------
