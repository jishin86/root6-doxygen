382 template <class Tmpl> TBuffer &operator>>(TBuffer &buf, Tmpl *&obj)
383 {
384  // Read TObject derived classes from a TBuffer. Need to provide
385  // custom version for non-TObject derived classes.
386 
387  // This operator has to be a templated and/or automatically
388  // generated if we want to be able to check the type of the
389  // incoming object. I.e. a operator>>(TBuffer &buf, TObject *&)
390  // would not be sufficient to pass the information 'which class do we want'
391  // since the pointer could be zero (so typeid(*obj) is not usable).
392 
393  auto cl = TClass::GetClass<Tmpl>();
394  obj = (Tmpl *) ( (void*) buf.ReadObjectAny(cl) );
395  return buf;
396 }
397 
398 template <class Tmpl> TBuffer &operator<<(TBuffer &buf, const Tmpl *obj)
399 {
400  auto cl = (obj) ? TClass::GetClass<Tmpl>() : nullptr;
401  buf.WriteObjectAny(obj, cl);
402  return buf;
403 }
