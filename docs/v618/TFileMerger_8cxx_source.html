18 It can be used to copy files (not only ROOT files), using TFile or
19 any of its remote file access plugins. It is therefore useful in

59 static const Int_t kCpProgress = BIT(14);

61 ////////////////////////////////////////////////////////////////////////////////
62 /// Return the maximum number of allowed opened files minus some wiggle room
63 /// for CINT or at least of the standard library (stdio).
64 
65 static Int_t R__GetSystemMaxOpenedFiles()
66 {
67  int maxfiles;
68 #ifdef WIN32
69  maxfiles = _getmaxstdio();
70 #else
71  rlimit filelimit;
72  if (getrlimit(RLIMIT_NOFILE,&filelimit)==0) {
73  maxfiles = filelimit.rlim_cur;
74  } else {
75  // We could not get the value from getrlimit, let's return a reasonable default.
76  maxfiles = 512;
77  }
78 #endif
79  if (maxfiles > kCintFileNumber) {
80  return maxfiles - kCintFileNumber;
81  } else if (maxfiles > 5) {
82  return maxfiles - 5;
83  } else {
84  return maxfiles;
85  }
86 }

126 /// Add file to file merger.
127 
128 Bool_t TFileMerger::AddFile(const char *url, Bool_t cpProgress)
129 {
130  if (fPrintLevel > 0) {
131  Printf("%s Source file %d: %s", fMsgPrefix.Data(), fFileList.GetEntries() + fExcessFiles.GetEntries() + 1, url);
132  }
133 
134  TFile *newfile = 0;
135  TString localcopy;
136 
137  if (fFileList.GetEntries() >= (fMaxOpenedFiles-1)) {
138 
139  TObjString *urlObj = new TObjString(url);
140  fMergeList.Add(urlObj);
141 
142  urlObj = new TObjString(url);
143  urlObj->SetBit(kCpProgress);
144  fExcessFiles.Add(urlObj);
145  return kTRUE;
146  }
147 
148  // We want gDirectory untouched by anything going on here
149  TDirectory::TContext ctxt;
150 
151  if (fLocal) {
152  TUUID uuid;
153  localcopy.Form("file:%s/ROOTMERGE-%s.root", gSystem->TempDirectory(), uuid.AsString());
154  if (!TFile::Cp(url, localcopy, cpProgress)) {
155  Error("AddFile", "cannot get a local copy of file %s", url);
156  return kFALSE;
157  }
158  newfile = TFile::Open(localcopy, "READ");
159  } else {
160  newfile = TFile::Open(url, "READ");
161  }
162 
163  // Zombie files should also be skipped
164  if (newfile && newfile->IsZombie()) {
165  delete newfile;
166  newfile = 0;
167  }
168 
169  if (!newfile) {
170  if (fLocal)
171  Error("AddFile", "cannot open local copy %s of URL %s",
172  localcopy.Data(), url);
173  else
174  Error("AddFile", "cannot open file %s", url);
175  return kFALSE;
176  } else {
177  if (fOutputFile && fOutputFile->GetCompressionLevel() != newfile->GetCompressionLevel()) fCompressionChange = kTRUE;
178 
179  newfile->SetBit(kCanDelete);
180  fFileList.Add(newfile);
181 
182  TObjString *urlObj = new TObjString(url);
183  fMergeList.Add(urlObj);
184 
185  return kTRUE;
186  }
187 }

214 ///
215 /// Return kTRUE if the addition was successful.
216 
217 Bool_t TFileMerger::AddFile(TFile *source, Bool_t own, Bool_t cpProgress)
218 {
219  if (source == 0 || source->IsZombie()) {
220  return kFALSE;
221  }
222 
223  if (fPrintLevel > 0) {
224  Printf("%s Source file %d: %s",fMsgPrefix.Data(),fFileList.GetEntries()+1,source->GetName());
225  }
226 
227  TFile *newfile = 0;
228  TString localcopy;
229 
230  // We want gDirectory untouched by anything going on here
231  TDirectory::TContext ctxt;
232  if (fLocal && !source->InheritsFrom(TMemFile::Class())) {
233  TUUID uuid;
234  localcopy.Form("file:%s/ROOTMERGE-%s.root", gSystem->TempDirectory(), uuid.AsString());
235  if (!source->Cp(localcopy, cpProgress)) {
236  Error("AddFile", "cannot get a local copy of file %s", source->GetName());
237  return kFALSE;
238  }
239  newfile = TFile::Open(localcopy, "READ");
240  // Zombie files should also be skipped
241  if (newfile && newfile->IsZombie()) {
242  delete newfile;
243  newfile = 0;
244  }
245  } else {
246  newfile = source;
247  }
248 
249  if (!newfile) {
250  if (fLocal)
251  Error("AddFile", "cannot open local copy %s of URL %s",
252  localcopy.Data(), source->GetName());
253  else
254  Error("AddFile", "cannot open file %s", source->GetName());
255  return kFALSE;
256  } else {
257  if (fOutputFile && fOutputFile->GetCompressionSettings() != newfile->GetCompressionSettings()) fCompressionChange = kTRUE;
258 
259  if (own || newfile != source) {
260  newfile->SetBit(kCanDelete);
261  } else {
262  newfile->ResetBit(kCanDelete);
263  }
264  fFileList.Add(newfile);
265 
266  TObjString *urlObj = new TObjString(source->GetName());
267  fMergeList.Add(urlObj);
268 
269  if (newfile != source && own) {
270  delete source;
271  }
272  return kTRUE;
273  }
274 }

375 ////////////////////////////////////////////////////////////////////////////////
376 /// Merge all objects in a directory
377 ///
378 /// The type is defined by the bit values in TFileMerger::EPartialMergeType.
379 
380 Bool_t TFileMerger::MergeRecursive(TDirectory *target, TList *sourcelist, Int_t type /* = kRegular | kAll */)
381 {
382  Bool_t status = kTRUE;
383  Bool_t onlyListed = kFALSE;
384  if (fPrintLevel > 0) {
385  Printf("%s Target path: %s",fMsgPrefix.Data(),target->GetPath());
386  }
387 
388  // Get the dir name
389  TString path(target->GetPath());
390  // coverity[unchecked_value] 'target' is from a file so GetPath always returns path starting with filename:
391  path.Remove(0, std::strlen(target->GetFile()->GetPath()));
392 
393  Int_t nguess = sourcelist->GetSize()+1000;
394  THashList allNames(nguess);
395  allNames.SetOwner(kTRUE);
396  // If the mode is set to skipping list objects, add names to the allNames list
397  if (type & kSkipListed) {
398  TObjArray *arr = fObjectNames.Tokenize(" ");
399  arr->SetOwner(kFALSE);
400  for (Int_t iname=0; iname<arr->GetEntriesFast(); iname++)
401  allNames.Add(arr->At(iname));
402  delete arr;
403  }
404  ((THashList*)target->GetList())->Rehash(nguess);
405  ((THashList*)target->GetListOfKeys())->Rehash(nguess);
406 
407  TFileMergeInfo info(target);
408  info.fIOFeatures = fIOFeatures;
409  info.fOptions = fMergeOptions;
410  if (fFastMethod && ((type&kKeepCompression) || !fCompressionChange) ) {
411  info.fOptions.Append(" fast");
412  }
413 
414  TFile *current_file;
415  TDirectory *current_sourcedir;
416  if (type & kIncremental) {
417  current_file = 0;
418  current_sourcedir = target;
419  } else {
420  current_file = (TFile*)sourcelist->First();
421  current_sourcedir = current_file->GetDirectory(path);
422  }
423  while (current_file || current_sourcedir) {
424  // When current_sourcedir != 0 and current_file == 0 we are going over the target
425  // for an incremental merge.
426  if (current_sourcedir && (current_file == 0 || current_sourcedir != target)) {
427 
428  // loop over all keys in this directory
429  TIter nextkey( current_sourcedir->GetListOfKeys() );
430  TKey *key;
431  TString oldkeyname;
432 
433  while ( (key = (TKey*)nextkey())) {
434 
435  // Keep only the highest cycle number for each key for mergeable objects. They are stored
436  // in the (hash) list consecutively and in decreasing order of cycles, so we can continue
437  // until the name changes. We flag the case here and we act consequently later.
438  Bool_t alreadyseen = (oldkeyname == key->GetName()) ? kTRUE : kFALSE;
439 
440  // Read in but do not copy directly the processIds.
441  if (strcmp(key->GetClassName(),"TProcessID") == 0) { key->ReadObj(); continue;}
442 
443  // If we have already seen this object [name], we already processed
444  // the whole list of files for this objects and we can just skip it
445  // and any related cycles.
446  if (allNames.FindObject(key->GetName())) {
447  oldkeyname = key->GetName();
448  continue;
449  }
450 
451  TClass *cl = TClass::GetClass(key->GetClassName());
452  if (!cl) {
453  Info("MergeRecursive", "cannot indentify object type (%s), name: %s title: %s",
454  key->GetClassName(), key->GetName(), key->GetTitle());
455  continue;
456  }
457  // For mergeable objects we add the names in a local hashlist handling them
458  // again (see above)
459  if (cl->GetMerge() || cl->InheritsFrom(TDirectory::Class()) ||
460  (cl->IsTObject() &&
461  (cl->GetMethodWithPrototype("Merge", "TCollection*,TFileMergeInfo*") ||
462  cl->GetMethodWithPrototype("Merge", "TCollection*"))))
463  allNames.Add(new TObjString(key->GetName()));
464 
465  if (fNoTrees && cl->InheritsFrom(R__TTree_Class)) {
466  // Skip the TTree objects and any related cycles.
467  oldkeyname = key->GetName();
468  continue;
469  }
470  // Check if only the listed objects are to be merged
471  if (type & kOnlyListed) {
472  onlyListed = kFALSE;
473  oldkeyname = key->GetName();
474  oldkeyname += " ";
475  onlyListed = fObjectNames.Contains(oldkeyname);
476  oldkeyname = key->GetName();
477  if ((!onlyListed) && (!cl->InheritsFrom(TDirectory::Class()))) continue;
478  }
479 
480  if (!(type&kResetable && type&kNonResetable)) {
481  // If neither or both are requested at the same time, we merger both types.
482  if (!(type&kResetable)) {
483  if (cl->GetResetAfterMerge()) {
484  // Skip the object with a reset after merge routine (TTree and other incrementally mergeable objects)
485  oldkeyname = key->GetName();
486  continue;
487  }
488  }
489  if (!(type&kNonResetable)) {
490  if (!cl->GetResetAfterMerge()) {
491  // Skip the object without a reset after merge routine (Histograms and other non incrementally mergeable objects)
492  oldkeyname = key->GetName();
493  continue;
494  }
495  }
496  }
497  // read object from first source file
498  TObject *obj;
499  if (type & kIncremental) {
500  obj = current_sourcedir->GetList()->FindObject(key->GetName());
501  if (!obj) {
502  obj = key->ReadObj();
503  }
504  } else {
505  obj = key->ReadObj();
506  }
507  if (!obj) {
508  Info("MergeRecursive", "could not read object for key {%s, %s}",
509  key->GetName(), key->GetTitle());
510  continue;
511  }
512  if (cl->IsTObject() && cl != obj->IsA()) {
513  Error("MergeRecursive", "TKey and object retrieve disagree on type (%s vs %s). Continuing with %s.",
514  key->GetClassName(), obj->IsA()->GetName(), obj->IsA()->GetName());
515  cl = obj->IsA();
516  }
517  Bool_t canBeMerged = kTRUE;
518 
519  if ( cl->InheritsFrom( TDirectory::Class() ) ) {
520  // it's a subdirectory
521 
522  target->cd();
523  TDirectory *newdir;
524 
525  // For incremental or already seen we may have already a directory created
526  if (type & kIncremental || alreadyseen) {
527  newdir = target->GetDirectory(obj->GetName());
528  if (!newdir) {
529  newdir = target->mkdir( obj->GetName(), obj->GetTitle() );
530  }
531  } else {
532  newdir = target->mkdir( obj->GetName(), obj->GetTitle() );
533  }
534 
535  // newdir is now the starting point of another round of merging
536  // newdir still knows its depth within the target file via
537  // GetPath(), so we can still figure out where we are in the recursion
538 
539  // If this folder is a onlyListed object, merge everything inside.
540  if (onlyListed) type &= ~kOnlyListed;
541  status = MergeRecursive(newdir, sourcelist, type);
542  if (onlyListed) type |= kOnlyListed;
543  if (!status) return status;
544  } else if (cl->GetMerge()) {
545 
546  // Check if already treated
547  if (alreadyseen) continue;
548 
549  TList inputs;
550  Bool_t oneGo = fHistoOneGo && cl->InheritsFrom(R__TH1_Class);
551 
552  // Loop over all source files and merge same-name object
553  TFile *nextsource = current_file ? (TFile*)sourcelist->After( current_file ) : (TFile*)sourcelist->First();
554  if (nextsource == 0) {
555  // There is only one file in the list
556  ROOT::MergeFunc_t func = cl->GetMerge();
557  func(obj, &inputs, &info);
558  info.fIsFirst = kFALSE;
559  } else {
560  do {
561  // make sure we are at the correct directory level by cd'ing to path
562  TDirectory *ndir = nextsource->GetDirectory(path);
563  if (ndir) {
564  ndir->cd();
565  TKey *key2 = (TKey*)ndir->GetListOfKeys()->FindObject(key->GetName());
566  if (key2) {
567  TObject *hobj = key2->ReadObj();
568  if (!hobj) {
569  Info("MergeRecursive", "could not read object for key {%s, %s}; skipping file %s",
570  key->GetName(), key->GetTitle(), nextsource->GetName());
571  nextsource = (TFile*)sourcelist->After(nextsource);
572  continue;
573  }
574  // Set ownership for collections
575  if (hobj->InheritsFrom(TCollection::Class())) {
576  ((TCollection*)hobj)->SetOwner();
577  }
578  hobj->ResetBit(kMustCleanup);
579  inputs.Add(hobj);
580  if (!oneGo) {
581  ROOT::MergeFunc_t func = cl->GetMerge();
582  Long64_t result = func(obj, &inputs, &info);
583  info.fIsFirst = kFALSE;
584  if (result < 0) {
585  Error("MergeRecursive", "calling Merge() on '%s' with the corresponding object in '%s'",
586  obj->GetName(), nextsource->GetName());
587  }
588  inputs.Delete();
589  }
590  }
591  }
592  nextsource = (TFile*)sourcelist->After( nextsource );
593  } while (nextsource);
594  // Merge the list, if still to be done
595  if (oneGo || info.fIsFirst) {
596  ROOT::MergeFunc_t func = cl->GetMerge();
597  func(obj, &inputs, &info);
598  info.fIsFirst = kFALSE;
599  inputs.Delete();
600  }
601  }
602  } else if (cl->IsTObject() &&
603  cl->GetMethodWithPrototype("Merge", "TCollection*,TFileMergeInfo*") ) {
604  // Object implements Merge(TCollection*,TFileMergeInfo*) and has a reflex dictionary ...
605 
606  // Check if already treated
607  if (alreadyseen) continue;
608 
609  TList listH;
610  TString listHargs;
611  listHargs.Form("(TCollection*)0x%lx,(TFileMergeInfo*)0x%lx", (ULong_t)&listH,(ULong_t)&info);
612 
613  // Loop over all source files and merge same-name object
614  TFile *nextsource = current_file ? (TFile*)sourcelist->After( current_file ) : (TFile*)sourcelist->First();
615  if (nextsource == 0) {
616  // There is only one file in the list
617  Int_t error = 0;
618  obj->Execute("Merge", listHargs.Data(), &error);
619  info.fIsFirst = kFALSE;
620  if (error) {
621  Error("MergeRecursive", "calling Merge() on '%s' with the corresponding object in '%s'",
622  obj->GetName(), key->GetName());
623  }
624  } else {
625  while (nextsource) {
626  // make sure we are at the correct directory level by cd'ing to path
627  TDirectory *ndir = nextsource->GetDirectory(path);
628  if (ndir) {
629  ndir->cd();
630  TKey *key2 = (TKey*)ndir->GetListOfKeys()->FindObject(key->GetName());
631  if (key2) {
632  TObject *hobj = key2->ReadObj();
633  if (!hobj) {
634  Info("MergeRecursive", "could not read object for key {%s, %s}; skipping file %s",
635  key->GetName(), key->GetTitle(), nextsource->GetName());
636  nextsource = (TFile*)sourcelist->After(nextsource);
637  continue;
638  }
639  // Set ownership for collections
640  if (hobj->InheritsFrom(TCollection::Class())) {
641  ((TCollection*)hobj)->SetOwner();
642  }
643  hobj->ResetBit(kMustCleanup);
644  listH.Add(hobj);
645  Int_t error = 0;
646  obj->Execute("Merge", listHargs.Data(), &error);
647  info.fIsFirst = kFALSE;
648  if (error) {
649  Error("MergeRecursive", "calling Merge() on '%s' with the corresponding object in '%s'",
650  obj->GetName(), nextsource->GetName());
651  }
652  listH.Delete();
653  }
654  }
655  nextsource = (TFile*)sourcelist->After( nextsource );
656  }
657  // Merge the list, if still to be done
658  if (info.fIsFirst) {
659  Int_t error = 0;
660  obj->Execute("Merge", listHargs.Data(), &error);
661  info.fIsFirst = kFALSE;
662  listH.Delete();
663  }
664  }
665  } else if (cl->IsTObject() &&
666  cl->GetMethodWithPrototype("Merge", "TCollection*") ) {
667  // Object implements Merge(TCollection*) and has a reflex dictionary ...
668 
669  // Check if already treated
670  if (alreadyseen) continue;
671 
672  TList listH;
673  TString listHargs;
674  listHargs.Form("((TCollection*)0x%lx)", (ULong_t)&listH);
675 
676  // Loop over all source files and merge same-name object
677  TFile *nextsource = current_file ? (TFile*)sourcelist->After( current_file ) : (TFile*)sourcelist->First();
678  if (nextsource == 0) {
679  // There is only one file in the list
680  Int_t error = 0;
681  obj->Execute("Merge", listHargs.Data(), &error);
682  if (error) {
683  Error("MergeRecursive", "calling Merge() on '%s' with the corresponding object in '%s'",
684  obj->GetName(), key->GetName());
685  }
686  } else {
687  while (nextsource) {
688  // make sure we are at the correct directory level by cd'ing to path
689  TDirectory *ndir = nextsource->GetDirectory(path);
690  if (ndir) {
691  ndir->cd();
692  TKey *key2 = (TKey*)ndir->GetListOfKeys()->FindObject(key->GetName());
693  if (key2) {
694  TObject *hobj = key2->ReadObj();
695  if (!hobj) {
696  Info("MergeRecursive", "could not read object for key {%s, %s}; skipping file %s",
697  key->GetName(), key->GetTitle(), nextsource->GetName());
698  nextsource = (TFile*)sourcelist->After(nextsource);
699  continue;
700  }
701  // Set ownership for collections
702  if (hobj->InheritsFrom(TCollection::Class())) {
703  ((TCollection*)hobj)->SetOwner();
704  }
705  hobj->ResetBit(kMustCleanup);
706  listH.Add(hobj);
707  Int_t error = 0;
708  obj->Execute("Merge", listHargs.Data(), &error);
709  info.fIsFirst = kFALSE;
710  if (error) {
711  Error("MergeRecursive", "calling Merge() on '%s' with the corresponding object in '%s'",
712  obj->GetName(), nextsource->GetName());
713  }
714  listH.Delete();
715  }
716  }
717  nextsource = (TFile*)sourcelist->After( nextsource );
718  }
719  // Merge the list, if still to be done
720  if (info.fIsFirst) {
721  Int_t error = 0;
722  obj->Execute("Merge", listHargs.Data(), &error);
723  info.fIsFirst = kFALSE;
724  listH.Delete();
725  }
726  }
727  } else {
728  // Object is of no type that we can merge
729  canBeMerged = kFALSE;
730  }
731 
732  // now write the merged histogram (which is "in" obj) to the target file
733  // note that this will just store obj in the current directory level,
734  // which is not persistent until the complete directory itself is stored
735  // by "target->SaveSelf()" below
736  target->cd();
737 
738  oldkeyname = key->GetName();
739  //!!if the object is a tree, it is stored in globChain...
740  if(cl->InheritsFrom( TDirectory::Class() )) {
741  //printf("cas d'une directory\n");
742  // Do not delete the directory if it is part of the output
743  // and we are in incremental mode (because it will be reuse
744  // and has not been written to disk (for performance reason).
745  // coverity[var_deref_model] the IsA()->InheritsFrom guarantees that the dynamic_cast will succeed.
746  if (!(type&kIncremental) || dynamic_cast<TDirectory*>(obj)->GetFile() != target) {
747  delete obj;
748  }
749  } else if (cl->InheritsFrom( TCollection::Class() )) {
750  // Don't overwrite, if the object were not merged.
751  if ( obj->Write( oldkeyname, canBeMerged ? TObject::kSingleKey | TObject::kOverwrite : TObject::kSingleKey) <= 0 ) {
752  status = kFALSE;
753  }
754  ((TCollection*)obj)->SetOwner();
755  delete obj;
756  } else {
757  // Don't overwrite, if the object were not merged.
758  // NOTE: this is probably wrong for emulated objects.
759  if (cl->IsTObject()) {
760  if ( obj->Write( oldkeyname, canBeMerged ? TObject::kOverwrite : 0) <= 0) {
761  status = kFALSE;
762  }
763  } else {
764  if ( target->WriteObjectAny( (void*)obj, cl, oldkeyname, canBeMerged ? "OverWrite" : "" ) <= 0) {
765  status = kFALSE;
766  }
767  }
768  cl->Destructor(obj); // just in case the class is not loaded.
769  }
770  info.Reset();
771  } // while ( ( TKey *key = (TKey*)nextkey() ) )
772  }
773  current_file = current_file ? (TFile*)sourcelist->After(current_file) : (TFile*)sourcelist->First();
774  if (current_file) {
775  current_sourcedir = current_file->GetDirectory(path);
776  } else {
777  current_sourcedir = 0;
778  }
779  }
780  // save modifications to the target directory.
781  if (!(type&kIncremental)) {
782  // In case of incremental build, we will call Write on the top directory/file, so we do not need
783  // to call SaveSelf explicilty.
784  target->SaveSelf(kTRUE);
785  }
786  return status;
787 }

789 ////////////////////////////////////////////////////////////////////////////////
790 /// Merge the files. If no output file was specified it will write into
791 /// the file "FileMerger.root" in the working directory. Returns true
792 /// on success, false in case of error.
793 /// The type is defined by the bit values in EPartialMergeType:
794 /// kRegular : normal merge, overwritting the output file
795 /// kIncremental : merge the input file with the content of the output file (if already exising) (default)
796 /// kAll : merge all type of objects (default)
797 /// kResetable : merge only the objects with a MergeAfterReset member function.
798 /// kNonResetable : merge only the objects without a MergeAfterReset member function.
799 ///
800 /// If the type is set to kIncremental the output file is done deleted at the end of
801 /// this operation. If the type is not set to kIncremental, the output file is closed.
802 
803 Bool_t TFileMerger::PartialMerge(Int_t in_type)
804 {
805  if (!fOutputFile) {
806  TString outf(fOutputFilename);
807  if (outf.IsNull()) {
808  outf.Form("file:%s/FileMerger.root", gSystem->TempDirectory());
809  Info("PartialMerge", "will merge the results to the file %s\n"
810  "since you didn't specify a merge filename",
811  TUrl(outf).GetFile());
812  }
813  if (!OutputFile(outf.Data())) {
814  return kFALSE;
815  }
816  }
817 
818  // Special treament for the single file case ...
819  if ((fFileList.GetEntries() == 1) && !fExcessFiles.GetEntries() &&
820  !(in_type & kIncremental) && !fCompressionChange && !fExplicitCompLevel) {
821  fOutputFile->Close();
822  SafeDelete(fOutputFile);
823 
824  TFile *file = (TFile *) fFileList.First();
825  if (!file || (file && file->IsZombie())) {
826  Error("PartialMerge", "one-file case: problem attaching to file");
827  return kFALSE;
828  }
829  Bool_t result = kTRUE;
830  if (!(result = file->Cp(fOutputFilename))) {
831  Error("PartialMerge", "one-file case: could not copy '%s' to '%s'",
832  file->GetPath(), fOutputFilename.Data());
833  return kFALSE;
834  }
835  if (file->TestBit(kCanDelete)) file->Close();
836 
837  // Remove the temporary file
838  if (fLocal && !file->InheritsFrom(TMemFile::Class())) {
839  TUrl u(file->GetPath(), kTRUE);
840  if (gSystem->Unlink(u.GetFile()) != 0)
841  Warning("PartialMerge", "problems removing temporary local file '%s'", u.GetFile());
842  }
843  fFileList.Clear();
844  return result;
845  }
846 
847  fOutputFile->SetBit(kMustCleanup);
848 
849  TDirectory::TContext ctxt;
850 
851  Bool_t result = kTRUE;
852  Int_t type = in_type;
853  while (result && fFileList.GetEntries()>0) {
854  result = MergeRecursive(fOutputFile, &fFileList, type);
855 
856  // Remove local copies if there are any
857  TIter next(&fFileList);
858  TFile *file;
859  while ((file = (TFile*) next())) {
860  // close the files
861  if (file->TestBit(kCanDelete)) file->Close();
862  // remove the temporary files
863  if(fLocal && !file->InheritsFrom(TMemFile::Class())) {
864  TString p(file->GetPath());
865  // coverity[unchecked_value] Index is return a value with range or NPos to select the whole name.
866  p = p(0, p.Index(':',0));
867  gSystem->Unlink(p);
868  }
869  }
870  fFileList.Clear();
871  if (result && fExcessFiles.GetEntries() > 0) {
872  // We merge the first set of files in the output,
873  // we now need to open the next set and make
874  // sure we accumulate into the output, so we
875  // switch to incremental merging (if not already set)
876  type = type | kIncremental;
877  result = OpenExcessFiles();
878  }
879  }
880  if (!result) {
881  Error("Merge", "error during merge of your ROOT files");
882  } else {
883  // Close or write is required so the file is complete.
884  if (in_type & kIncremental) {
885  fOutputFile->Write("",TObject::kOverwrite);
886  } else {
887  fOutputFile->Close();
888  }
889  }
890 
891  // Cleanup
892  if (in_type & kIncremental) {
893  Clear();
894  } else {
895  fOutputFile->ResetBit(kMustCleanup);
896  SafeDelete(fOutputFile);
897  }
898  return result;
899 }
