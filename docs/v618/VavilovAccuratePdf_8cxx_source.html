89 double VavilovAccuratePdf::DoEval(double x) const {
90  VavilovAccurate v(fP[3], fP[4]);

94 double VavilovAccuratePdf::DoEvalPar(double x, const double * p) const {
95  if (!p) return 0;
96  // p[0]: norm, p[1]: x0, p[2]: width, p[3]: kappa, p[4]: beta2
