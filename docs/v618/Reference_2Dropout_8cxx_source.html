29 void TReference<Real_t>::Dropout(TMatrixT<Real_t> & B, Real_t dropoutProbability)
30 {
31  size_t m,n;
32  m = B.GetNrows();
33  n = B.GetNcols();
34 
35  TRandom rand(time(nullptr));
36 
37  for (size_t i = 0; i < m; i++) {
38  for (size_t j = 0; j < n; j++) {
39  Real_t r = rand.Uniform();
40  if (r >= dropoutProbability) {
41  B(i,j) = 0.0;
42  } else {
43  B(i,j) /= dropoutProbability;
44  }
45  }
46  }
47 }
