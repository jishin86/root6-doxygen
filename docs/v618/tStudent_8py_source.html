41 xx = []
42 xx.append(-1.5)

44  xx.append(quant.GetBinContent(i))
45 xx.append(1.5)
46 
47 pdfq = []
48 for i in range(9):

50  name = "pdf%d" % i

52  for j in range(1, nbin):

54  pdfq[i].SetBinContent(j, ROOT.Math.tdistribution_pdf(x, 3))
