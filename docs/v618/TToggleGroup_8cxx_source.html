99 ////////////////////////////////////////////////////////////////////////////////
100 /// Selector a toggle.
101 
102 void TToggleGroup::Select(TToggle *t)
103 {
104  TIter next(fToggles);
105  TToggle *i = 0;
106 
107  // Untoggle toggled , and toggle this one if it's present on a list!
108 
109  while ((i = (TToggle*)next()))
110  if ( i->GetState() || (i==t) )
111  i->Toggle();
112 }
