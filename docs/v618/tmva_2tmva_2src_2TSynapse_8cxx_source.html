96 ////////////////////////////////////////////////////////////////////////////////

98 
99 void TMVA::TSynapse::AdjustWeight()
100 {
101  Double_t wDelta = fDelta / fCount;
102  fWeight += -fLearnRate * wDelta;
103  InitDelta();
104 }
