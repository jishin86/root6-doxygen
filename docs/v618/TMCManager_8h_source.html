112  /// Apply something to all engines
113  template <typename F>
114  void Apply(F engineLambda)
115  {
116  for (auto &mc : fEngines) {
117  // We never know whether static method TVirtualMC::GetMC() is used in any way so update before calling the
118  // lambda.
119  UpdateEnginePointers(mc);
120  engineLambda(mc);
121  }
122  }
123 
124  /// Initialize engines
125  void Init();
126  /// Further specific initialization
127  template <typename F>
128  void Init(F initFunction)
129  {
130  if (fIsInitializedUser) {
131  return;
132  }
133  Init();
134  for (auto &mc : fEngines) {
135  // Set to current engine and call user init procedure
136  UpdateEnginePointers(mc);
137  initFunction(mc);
138  }
139  fIsInitializedUser = kTRUE;
140  }
