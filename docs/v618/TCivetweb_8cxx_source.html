79 int websocket_connect_handler(const struct mg_connection *conn, void *)
80 {
81  const struct mg_request_info *request_info = mg_get_request_info(conn);
82  if (!request_info)
83  return 1;
84 
85  TCivetweb *engine = (TCivetweb *)request_info->user_data;
86  if (!engine || engine->IsTerminating())
87  return 1;
88  THttpServer *serv = engine->GetServer();
89  if (!serv)
90  return 1;
91 
92  auto arg = std::make_shared<THttpCallArg>();
93  arg->SetPathAndFileName(request_info->local_uri); // path and file name
94  arg->SetQuery(request_info->query_string); // query arguments
95  arg->SetWSId(TString::Hash((void *)&conn, sizeof(void *)));
96  arg->SetMethod("WS_CONNECT");
97 
98  Bool_t execres = serv->ExecuteWS(arg, kTRUE, kTRUE);
99 
100  return execres && !arg->Is404() ? 0 : 1;
101 }

129 int websocket_data_handler(struct mg_connection *conn, int code, char *data, size_t len, void *)
130 {
131  const struct mg_request_info *request_info = mg_get_request_info(conn);
132 
133  // do not handle empty data
134  if (len == 0)
135  return 1;
136 
137  TCivetweb *engine = (TCivetweb *)request_info->user_data;
138  if (!engine || engine->IsTerminating())
139  return 1;
140  THttpServer *serv = engine->GetServer();
141  if (!serv)
142  return 1;
143 
144  std::string *conn_data = (std::string *) mg_get_user_connection_data(conn);
145 
146  // this is continuation of the request
147  if (!(code & 0x80)) {
148  if (!conn_data) {
149  conn_data = new std::string(data,len);
150  mg_set_user_connection_data(conn, conn_data);
151  } else {
152  conn_data->append(data,len);
153  }
154  return 1;
155  }
156 
157  auto arg = std::make_shared<THttpCallArg>();
158  arg->SetPathAndFileName(request_info->local_uri); // path and file name
159  arg->SetQuery(request_info->query_string); // query arguments
160  arg->SetWSId(TString::Hash((void *)&conn, sizeof(void *)));
161  arg->SetMethod("WS_DATA");
162 
163  if (conn_data) {
164  mg_set_user_connection_data(conn, nullptr);
165  conn_data->append(data,len);
166  arg->SetPostData(std::move(*conn_data));
167  delete conn_data;
168  } else {
169  arg->SetPostData(std::string(data,len));
170  }
171 
172  serv->ExecuteWS(arg, kTRUE, kTRUE);
173 
174  return 1;
175 }

219 static int begin_request_handler(struct mg_connection *conn, void *)
220 {
221  const struct mg_request_info *request_info = mg_get_request_info(conn);
222 
223  TCivetweb *engine = (TCivetweb *)request_info->user_data;
224  if (!engine || engine->IsTerminating())
225  return 0;
226  THttpServer *serv = engine->GetServer();
227  if (!serv)
228  return 0;
229 
230  auto arg = std::make_shared<THttpCallArg>();
231 
232  TString filename;
233 
234  Bool_t execres = kTRUE, debug = engine->IsDebugMode();
235 
236  if (!debug && serv->IsFileRequested(request_info->local_uri, filename)) {
237  if ((filename.Index(".js") != kNPOS) || (filename.Index(".css") != kNPOS)) {
238  std::string buf = THttpServer::ReadFileContent(filename.Data());
239  if (buf.empty()) {
240  arg->Set404();
241  } else {
242  arg->SetContentType(THttpServer::GetMimeType(filename.Data()));
243  arg->SetContent(std::move(buf));
244  if (engine->GetMaxAge() > 0)
245  arg->AddHeader("Cache-Control", TString::Format("max-age=%d", engine->GetMaxAge()));
246  else
247  arg->AddNoCacheHeader();
248  arg->SetZipping();
249  }
250  } else {
251  arg->SetFile(filename.Data());
252  }
253  } else {
254  arg->SetPathAndFileName(request_info->local_uri); // path and file name
255  arg->SetQuery(request_info->query_string); // query arguments
256  arg->SetTopName(engine->GetTopName());
257  arg->SetMethod(request_info->request_method); // method like GET or POST
258  if (request_info->remote_user)
259  arg->SetUserName(request_info->remote_user);
260 
261  TString header;
262  for (int n = 0; n < request_info->num_headers; n++)
263  header.Append(
264  TString::Format("%s: %s\r\n", request_info->http_headers[n].name, request_info->http_headers[n].value));
265  arg->SetRequestHeader(header);
266 
267  const char *len = mg_get_header(conn, "Content-Length");
268  Int_t ilen = len ? TString(len).Atoi() : 0;
269 
270  if (ilen > 0) {
271  std::string buf;
272  buf.resize(ilen);
273  Int_t iread = mg_read(conn, (void *) buf.data(), ilen);
274  if (iread == ilen)
275  arg->SetPostData(std::move(buf));
276  }
277 
278  if (debug) {
279  TString cont;
280  cont.Append("<title>Civetweb echo</title>");
281  cont.Append("<h1>Civetweb echo</h1>\n");
282 
283  static int count = 0;
284 
285  cont.Append(TString::Format("Request %d:<br/>\n<pre>\n", ++count));
286  cont.Append(TString::Format(" Method : %s\n", arg->GetMethod()));
287  cont.Append(TString::Format(" PathName : %s\n", arg->GetPathName()));
288  cont.Append(TString::Format(" FileName : %s\n", arg->GetFileName()));
289  cont.Append(TString::Format(" Query : %s\n", arg->GetQuery()));
290  cont.Append(TString::Format(" PostData : %ld\n", arg->GetPostDataLength()));
291  if (arg->GetUserName())
292  cont.Append(TString::Format(" User : %s\n", arg->GetUserName()));
293 
294  cont.Append("</pre><p>\n");
295 
296  cont.Append("Environment:<br/>\n<pre>\n");
297  for (int n = 0; n < request_info->num_headers; n++)
298  cont.Append(
299  TString::Format(" %s = %s\n", request_info->http_headers[n].name, request_info->http_headers[n].value));
300  cont.Append("</pre><p>\n");
301 
302  arg->SetContentType("text/html");
303 
304  arg->SetContent(cont);
305 
306  } else {
307  execres = serv->ExecuteHttp(arg);
308  }
309  }
310 
311  if (!execres || arg->Is404()) {
312  std::string hdr = arg->FillHttpHeader("HTTP/1.1");
313  mg_printf(conn, "%s", hdr.c_str());
314  } else if (arg->IsFile()) {
315  mg_send_file(conn, (const char *)arg->GetContent());
316  } else {
317 
318  Bool_t dozip = kFALSE;
319  switch (arg->GetZipping()) {
320  case THttpCallArg::kNoZip: dozip = kFALSE; break;
321  case THttpCallArg::kZipLarge:
322  if (arg->GetContentLength() < 10000) break;
323  case THttpCallArg::kZip:
324  // check if request header has Accept-Encoding
325  for (int n = 0; n < request_info->num_headers; n++) {
326  TString name = request_info->http_headers[n].name;
327  if (name.Index("Accept-Encoding", 0, TString::kIgnoreCase) != 0)
328  continue;
329  TString value = request_info->http_headers[n].value;
330  dozip = (value.Index("gzip", 0, TString::kIgnoreCase) != kNPOS);
331  break;
332  }
333 
334  break;
335  case THttpCallArg::kZipAlways: dozip = kTRUE; break;
336  }
337 
338  if (dozip)
339  arg->CompressWithGzip();
340 
341  std::string hdr = arg->FillHttpHeader("HTTP/1.1");
342  mg_printf(conn, "%s", hdr.c_str());
343 
344  if (arg->GetContentLength() > 0)
345  mg_write(conn, arg->GetContent(), (size_t)arg->GetContentLength());
346  }
347 
348  // Returning non-zero tells civetweb that our function has replied to
349  // the client, and civetweb should not send client any more data.
350  return 1;
351 }

353 //////////////////////////////////////////////////////////////////////////
354 // //
355 // TCivetweb //
356 // //
357 // http server implementation, based on civetweb embedded server //
358 // It is default kind of engine, created for THttpServer //
359 // Currently v1.8 from https://github.com/civetweb/civetweb is used //
360 // //
361 // Following additional options can be specified: //
362 // top=foldername - name of top folder, seen in the browser //
363 // thrds=N - use N threads to run civetweb server (default 5) //
364 // auth_file - global authentication file //
365 // auth_domain - domain name, used for authentication //
366 // //
367 // Example: //
368 // new THttpServer("http:8080?top=MyApp&thrds=3"); //
369 // //
370 // Authentication: //
371 // When auth_file and auth_domain parameters are specified, access //
372 // to running http server will be possible only after user //
373 // authentication, using so-call digest method. To generate //
374 // authentication file, htdigest routine should be used: //
375 // //
376 // [shell] htdigest -c .htdigest domain_name user //
377 // //
378 // When creating server, parameters should be: //
379 // //
380 // new THttpServer("http:8080?auth_file=.htdigets&auth_domain=domain_name"); //
381 // //

416 /// Creates embedded civetweb server
417 /// As main argument, http port should be specified like "8090".
418 /// Or one can provide combination of ipaddress and portnumber like 127.0.0.1:8090
419 /// Extra parameters like in URL string could be specified after '?' mark:
420 /// thrds=N - there N is number of threads used by the civetweb (default is 10)
421 /// top=name - configure top name, visible in the web browser
422 /// ssl_certificate=filename - SSL certificate, see docs/OpenSSL.md from civetweb
423 /// auth_file=filename - authentication file name, created with htdigets utility
424 /// auth_domain=domain - authentication domain
425 /// websocket_timeout=tm - set web sockets timeout in seconds (default 300)
426 /// websocket_disable - disable web sockets handling (default enabled)
427 /// bind - ip address to bind server socket
428 /// loopback - bind specified port to loopback 127.0.0.1 address
429 /// debug - enable debug mode, server always returns html page with request info
430 /// log=filename - configure civetweb log file
431 /// max_age=value - configures "Cache-Control: max_age=value" http header for all file-related requests, default 3600
432 /// nocache - try to fully disable cache control for file requests
433 /// Examples:
434 /// http:8080?websocket_disable
435 /// http:7546?thrds=30&websocket_timeout=20

437 Bool_t TCivetweb::Create(const char *args)
438 {
439  fCallbacks = malloc(sizeof(struct mg_callbacks));
440  memset(fCallbacks, 0, sizeof(struct mg_callbacks));
441  //((struct mg_callbacks *) fCallbacks)->begin_request = begin_request_handler;
442  ((struct mg_callbacks *)fCallbacks)->log_message = log_message_handler;
443  TString sport = IsSecured() ? "8480s" : "8080", num_threads = "10", websocket_timeout = "300000";
444  TString auth_file, auth_domain, log_file, ssl_cert, max_age;
445  Bool_t use_ws = kTRUE;
446 
447  // extract arguments
448  if (args && (strlen(args) > 0)) {
449 
450  // first extract port number
451  sport = "";
452  while ((*args != 0) && (*args != '?') && (*args != '/'))
453  sport.Append(*args++);
454  if (IsSecured() && (sport.Index("s")==kNPOS)) sport.Append("s");
455 
456  // than search for extra parameters
457  while ((*args != 0) && (*args != '?'))
458  args++;
459 
460  if (*args == '?') {
461  TUrl url(TString::Format("http://localhost/folder%s", args));
462 
463  if (url.IsValid()) {
464  url.ParseOptions();
465 
466  const char *top = url.GetValueFromOptions("top");
467  if (top)
468  fTopName = top;
469 
470  const char *log = url.GetValueFromOptions("log");
471  if (log)
472  log_file = log;
473 
474  Int_t thrds = url.GetIntValueFromOptions("thrds");
475  if (thrds > 0)
476  num_threads.Form("%d", thrds);
477 
478  const char *afile = url.GetValueFromOptions("auth_file");
479  if (afile)
480  auth_file = afile;
481 
482  const char *adomain = url.GetValueFromOptions("auth_domain");
483  if (adomain)
484  auth_domain = adomain;
485 
486  const char *sslc = url.GetValueFromOptions("ssl_cert");
487  if (sslc)
488  ssl_cert = sslc;
489 
490  Int_t wtmout = url.GetIntValueFromOptions("websocket_timeout");
491  if (wtmout > 0) {
492  websocket_timeout.Format("%d", wtmout * 1000);
493  use_ws = kTRUE;
494  }
495 
496  if (url.HasOption("websocket_disable"))
497  use_ws = kFALSE;
498 
499  if (url.HasOption("debug"))
500  fDebug = kTRUE;
501 
502  if (url.HasOption("loopback") && (sport.Index(":") == kNPOS))
503  sport = TString("127.0.0.1:") + sport;
504 
505  if (url.HasOption("bind") && (sport.Index(":") == kNPOS)) {
506  const char *addr = url.GetValueFromOptions("bind");
507  if (addr && strlen(addr))
508  sport = TString(addr) + ":" + sport;
509  }
510 
511  if (GetServer() && url.HasOption("cors")) {
512  const char *cors = url.GetValueFromOptions("cors");
513  GetServer()->SetCors(cors && *cors ? cors : "*");
514  }
515 
516  if (url.HasOption("nocache"))
517  fMaxAge = 0;
518 
519  if (url.HasOption("max_age"))
520  fMaxAge = url.GetIntValueFromOptions("max_age");
521 
522  max_age.Form("%d", fMaxAge);
523  }
524  }
525  }
526

528  int op(0);
529 
530  Info("Create", "Starting HTTP server on port %s", sport.Data());
531 
532  options[op++] = "listening_ports";
533  options[op++] = sport.Data();
534  options[op++] = "num_threads";
535  options[op++] = num_threads.Data();
536 
537  if (use_ws) {
538  options[op++] = "websocket_timeout_ms";
539  options[op++] = websocket_timeout.Data();
540  }
541 
542  if ((auth_file.Length() > 0) && (auth_domain.Length() > 0)) {
543  options[op++] = "global_auth_file";
544  options[op++] = auth_file.Data();
545  options[op++] = "authentication_domain";
546  options[op++] = auth_domain.Data();
547  }
548 
549  if (log_file.Length() > 0) {
550  options[op++] = "error_log_file";
551  options[op++] = log_file.Data();
552  }
553 
554  if (ssl_cert.Length() > 0) {
555  options[op++] = "ssl_certificate";
556  options[op++] = ssl_cert.Data();
557  } else if (IsSecured()) {
558  Error("Create", "No SSL certificate file configured");
559  }
560 
561  if (max_age.Length() > 0) {
562  options[op++] = "static_file_max_age";
563  options[op++] = max_age.Data();
564  }
565 
566  options[op++] = nullptr;
567 
568  // Start the web server.
569  fCtx = mg_start((struct mg_callbacks *)fCallbacks, this, options);
570 
571  if (!fCtx)
572  return kFALSE;
573 
574  mg_set_request_handler((struct mg_context *)fCtx, "/", begin_request_handler, nullptr);
575 
576  if (use_ws)
577  mg_set_websocket_handler((struct mg_context *)fCtx, "**root.websocket$", websocket_connect_handler,
578  websocket_ready_handler, websocket_data_handler, websocket_close_handler, nullptr);
579 
580  return kTRUE;
581 }
