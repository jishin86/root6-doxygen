22 double VectorUtil::Phi_0_2pi(double angle) {
23  // returns phi angle in the interval (0,2*PI]
24  if ( angle <= 2.*M_PI && angle > 0 ) return angle;
25 
26  if ( angle > 0 ) {
27  int n = static_cast<int>( angle/(2.*M_PI) );

34 }
