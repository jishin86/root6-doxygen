31 double SqrtLowParameterTransformation::Ext2int(double value, double lower, const MnMachinePrecision& ) const {
32  // external to internal transformation
33  double yy = value - lower + 1.;
34  double yy2 = yy*yy;
35  if (yy2 < 1. )
36  return 0;
37  else
38  return sqrt( yy2 -1);
