38 Int_t TSelectorScalar::Merge(TCollection *list)

43  while (TObject *obj = next()) {
44  TSelectorScalar *c = dynamic_cast<TSelectorScalar*>(obj);

46  Inc(c->GetVal());
47  n++;
48  }
49  }
50 
51  return n;
