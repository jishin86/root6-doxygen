46  } else {

57  // ----------------------------------------------------------------
58  //
59  // P.d.f. constructor arguments may by any type of RooAbsArg, but also
60  //
61  // Double_t --> converted to RooConst(...)
62  // {a,b,c} --> converted to RooArgSet() or RooArgList() depending on required ctor arg

65
