12 /**
13 \class TXMLNode
14 \ingroup IO
15 
16 TXMLNode contains a pointer to xmlNode, which is a node under the
17 DOM tree. A node can be an Element, an Attribute, a Text Node
18 or a Comment Node.
19 One can navigate the DOM tree by accessing the siblings and
20 parent or child nodes. Also retriving the Attribute or the Text in
21 an Element node.
22 */

104 ////////////////////////////////////////////////////////////////////////////////
105 /// Returns a list of node's attribute if any,
106 /// returns 0 if no attribute.
107 
108 TList *TXMLNode::GetAttributes()
109 {
110  if (fAttrList)
111  return fAttrList;
112 
113  if (!HasAttributes())
114  return 0;
115 
116  fAttrList = new TList();
117  xmlAttr *attr_node = fXMLNode->properties;
118  for (; attr_node; attr_node = attr_node->next) {
119  fAttrList->Add(new TXMLAttr((const char *) attr_node->name,
120  (const char *) attr_node->children->content));
121  }
122 
123  return fAttrList;
124 }
