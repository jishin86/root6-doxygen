17 /**
18 \file RooTreeDataStore.cxx
19 \class RooTreeDataStore
20 \ingroup Roofitcore
21 
22 RooTreeDataStore is a TTree-backed data storage. When a file is opened before
23 creating the data storage, the storage will be file-backed. This reduces memory
24 pressure because it allows storing the data in the file and reading it on demand.
25 For a completely memory-backed storage, which is faster than the file-backed storage,
26 RooVectorDataStore can be used.
27 
28 With tree-backed storage, the tree can be found in the file with the name
29 `RooTreeDataStore_name_title` for a dataset created as
30 `RooDataSet("name", "title", ...)`.
31 
32 \note A file needs to be opened **before** creating the data storage to enable file-backed
33 storage.
34 ```
35 TFile outputFile("filename.root", "RECREATE");
36 RooAbsData::setDefaultStorageType(RooAbsData::Tree);
37 RooDataSet mydata(...);
38 ```
39 
40 One can also change between TTree- and std::vector-backed storage using
41 RooAbsData::convertToTreeStore() and
42 RooAbsData::convertToVectorStore().
43 **/

447 /// Load values from tree 't' into this data collection, optionally
448 /// selecting events using 'select' RooFormulaVar
449 ///
450 /// The source tree 't' is first clone as not disturb its branch
451 /// structure when retrieving information from it.
452 
453 void RooTreeDataStore::loadValues(const TTree *t, const RooFormulaVar* select, const char* /*rangeName*/, Int_t /*nStart*/, Int_t /*nStop*/)
454 {
455  // Clone source tree
456  // WVE Clone() crashes on trees, CloneTree() crashes on tchains :-(
457 
458  // Change directory to memory dir before cloning tree to avoid ROOT errors
459  TString pwd(gDirectory->GetPath()) ;
460  TString memDir(gROOT->GetName()) ;
461  memDir.Append(":/") ;
462  Bool_t notInMemNow= (pwd!=memDir) ;
463 
464  if (notInMemNow) {
465  gDirectory->cd(memDir) ;
466  }
467 
468  TTree* tClone ;
469  if (dynamic_cast<const TChain*>(t)) {
470  tClone = (TTree*) t->Clone() ;
471  } else {
472  tClone = ((TTree*)t)->CloneTree() ;
473  }
474 
475  // Change directory back to original directory
476  tClone->SetDirectory(0) ;
477 
478  if (notInMemNow) {
479  gDirectory->cd(pwd) ;
480  }
481 
482  // Clone list of variables
483  RooArgSet *sourceArgSet = (RooArgSet*) _varsww.snapshot(kFALSE) ;
484 
485  // Attach args in cloned list to cloned source tree
486  TIterator* sourceIter = sourceArgSet->createIterator() ;
487  RooAbsArg* sourceArg = 0;
488  while ((sourceArg=(RooAbsArg*)sourceIter->Next())) {
489  sourceArg->attachToTree(*tClone,_defTreeBufSize) ;
490  }
491 
492  // Redirect formula servers to sourceArgSet
493  RooFormulaVar* selectClone(0) ;
494  if (select) {
495  selectClone = (RooFormulaVar*) select->cloneTree() ;
496  selectClone->recursiveRedirectServers(*sourceArgSet) ;
497  selectClone->setOperMode(RooAbsArg::ADirty,kTRUE) ;
498  }
499 
500  // Loop over events in source tree
501  RooAbsArg* destArg = 0;
502  TIterator* destIter = _varsww.createIterator() ;
503  Int_t numInvalid(0) ;
504  Int_t nevent= (Int_t)tClone->GetEntries();
505  for(Int_t i=0; i < nevent; ++i) {
506  Int_t entryNumber=tClone->GetEntryNumber(i);
507  if (entryNumber<0) break;
508  tClone->GetEntry(entryNumber,1);
509 
510  // Copy from source to destination
511  destIter->Reset() ;
512  sourceIter->Reset() ;
513  Bool_t allOK(kTRUE) ;
514  while ((destArg = (RooAbsArg*)destIter->Next())) {
515  sourceArg = (RooAbsArg*) sourceIter->Next() ;
516  destArg->copyCache(sourceArg) ;
517  sourceArg->copyCache(destArg) ;
518  if (!destArg->isValid()) {
519  numInvalid++ ;
520  allOK=kFALSE ;
521  break ;
522  }
523  }
524 
525  // Does this event pass the cuts?
526  if (!allOK || (selectClone && selectClone->getVal()==0)) {
527  continue ;
528  }
529 
530  fill() ;
531  }
532  delete destIter ;
533 
534  if (numInvalid>0) {
535  coutI(Eval) << "RooTreeDataStore::loadValues(" << GetName() << ") Ignored " << numInvalid << " out of range events" << endl ;
536  }
537 
538  SetTitle(t->GetTitle());
539 
540  delete sourceIter ;
541  delete sourceArgSet ;
542  delete selectClone ;
543  delete tClone ;
544 }

552 /// Load values from dataset 't' into this data collection, optionally
553 /// selecting events using 'select' RooFormulaVar
554 ///

556 void RooTreeDataStore::loadValues(const RooAbsDataStore *ads, const RooFormulaVar* select,
557  const char* rangeName, Int_t nStart, Int_t nStop)
558 {

560  RooFormulaVar* selectClone(0) ;
561  if (select) {
562  selectClone = (RooFormulaVar*) select->cloneTree() ;
563  selectClone->recursiveRedirectServers(*ads->get()) ;
564  selectClone->setOperMode(RooAbsArg::ADirty,kTRUE) ;
565  }
566 
567  // Force RDS internal initialization
568  ads->get(0) ;
569 
570  // Loop over events in source tree
571  RooAbsArg* arg = 0;
572  TIterator* destIter = _varsww.createIterator() ;
573  Int_t nevent = nStop < ads->numEntries() ? nStop : ads->numEntries() ;
574  Bool_t allValid ;
575 
576  Bool_t isTDS = dynamic_cast<const RooTreeDataStore*>(ads) ;
577  if (isTDS) {
578  ((RooTreeDataStore*)(ads))->resetBuffers() ;
579  }
580  for(Int_t i=nStart; i < nevent ; ++i) {
581  ads->get(i) ;
582 
583  // Does this event pass the cuts?
584  if (selectClone && selectClone->getVal()==0) {
585  continue ;
586  }
587 
588 
589  if (isTDS) {
590  _varsww.assignValueOnly(((RooTreeDataStore*)ads)->_varsww) ;
591  } else {
592  _varsww.assignValueOnly(*ads->get()) ;
593  }
594 
595  destIter->Reset() ;
596  // Check that all copied values are valid
597  allValid=kTRUE ;
598  while((arg=(RooAbsArg*)destIter->Next())) {
599  if (!arg->isValid() || (rangeName && !arg->inRange(rangeName))) {
600  //cout << "arg " << arg->GetName() << " is not valid" << endl ;
601  //arg->Print("v") ;
602  allValid=kFALSE ;
603  break ;
604  }
605  }
606  //cout << "RooTreeData::loadValues(" << GetName() << ") allValid = " << (allValid?"T":"F") << endl ;
607  if (!allValid) {
608  continue ;
609  }
610 
611  _cachedVars = ((RooTreeDataStore*)ads)->_cachedVars ;
612  fill() ;
613  }
614  delete destIter ;
615  if (isTDS) {
616  ((RooTreeDataStore*)(ads))->restoreAlternateBuffers() ;
617  }
618 
619  delete selectClone ;
620  SetTitle(ads->GetTitle());
621 }
