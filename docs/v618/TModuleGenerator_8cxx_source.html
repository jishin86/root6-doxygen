92 /// Check whether the file's extension is compatible with C or C++.
93 /// Return whether source, header, Linkdef or nothing.
94 
95 TModuleGenerator::ESourceFileKind
96 TModuleGenerator::GetSourceFileKind(const char *filename) const
97 {
98  if (filename[0] == '-') return kSFKNotC;
99 
100  if (ROOT::TMetaUtils::IsLinkdefFile(filename)) {
101  return kSFKLinkdef;
102  }
103 
104  const size_t len = strlen(filename);
105  const char *ext = filename + len - 1;
106  while (ext >= filename && *ext != '.') --ext;
107  if (ext < filename || *ext != '.') {
108  // This might still be a system header, let's double check
109  // via the FileManager.
110  clang::Preprocessor &PP = fCI->getPreprocessor();
111  clang::HeaderSearch &HdrSearch = PP.getHeaderSearchInfo();
112  const clang::DirectoryLookup *CurDir = 0;
113  const clang::FileEntry *hdrFileEntry
114  = HdrSearch.LookupFile(filename, clang::SourceLocation(),
115  true /*isAngled*/, 0 /*FromDir*/, CurDir,
116  clang::ArrayRef<std::pair<const clang::FileEntry*,
117  const clang::DirectoryEntry*>>(),
118  0 /*IsMapped*/, 0 /*SearchPath*/, 0 /*RelativePath*/,
119  0 /*RequestingModule*/, 0/*SuggestedModule*/);
120  if (hdrFileEntry) {
121  return kSFKHeader;
122  }
123  return kSFKNotC;
124  }
125  ++ext;
126  const size_t lenExt = filename + len - ext;
127 
128  ESourceFileKind ret = kSFKNotC;
129  switch (lenExt) {
130  case 1: {
131  const char last = toupper(filename[len - 1]);
132  if (last == 'H') ret = kSFKHeader;
133  else if (last == 'C') ret = kSFKSource;
134  break;
135  }
136  case 2: {
137  if (filename[len - 2] == 'h' && filename[len - 1] == 'h')
138  ret = kSFKHeader;
139  else if (filename[len - 2] == 'c' && filename[len - 1] == 'c')
140  ret = kSFKSource;
141  break;
142  }
143  case 3: {
144  const char last = filename[len - 1];
145  if ((last == 'x' || last == 'p')
146  && filename[len - 2] == last) {
147  if (filename[len - 3] == 'h') ret = kSFKHeader;
148  else if (filename[len - 3] == 'c') ret = kSFKSource;
149  }
150  }
151  } // switch extension length
152 
153  return ret;
154 }

157 static
158 std::pair<std::string, std::string> SplitPPDefine(const std::string &in)
159 {
160  std::string::size_type posEq = in.find('=');
161  // No equal found: define to 1
162  if (posEq == std::string::npos)
163  return std::make_pair(in, "1");
164 
165  // Equal found
166  return std::pair<std::string, std::string>
167  (in.substr(0, posEq), in.substr(posEq + 1, std::string::npos));
168 }

171 /// Parse -I -D -U headers.h SomethingLinkdef.h.
172 
173 void TModuleGenerator::ParseArgs(const std::vector<std::string> &args)
174 {
175  for (size_t iPcmArg = 1 /*skip argv0*/, nPcmArg = args.size();
176  iPcmArg < nPcmArg; ++iPcmArg) {
177  ESourceFileKind sfk = GetSourceFileKind(args[iPcmArg].c_str());
178  if (sfk == kSFKHeader || sfk == kSFKSource) {
179  fHeaders.push_back(args[iPcmArg]);
180  } else if (sfk == kSFKLinkdef) {
181  fLinkDefFile = args[iPcmArg];
182  } else if (sfk == kSFKNotC && args[iPcmArg][0] == '-') {
183  switch (args[iPcmArg][1]) {
184  case 'I':
185  if (args[iPcmArg] != "-I." && args[iPcmArg] != "-Iinclude") {
186  fCompI.push_back(args[iPcmArg].c_str() + 2);
187  }
188  break;
189  case 'D':
190  if (args[iPcmArg] != "-DTRUE=1" && args[iPcmArg] != "-DFALSE=0"
191  && args[iPcmArg] != "-DG__NOCINTDLL") {
192  fCompD.push_back(SplitPPDefine(args[iPcmArg].c_str() + 2));
193  }
194  break;
195  case 'U':
196  fCompU.push_back(args[iPcmArg].c_str() + 2);
197  break;
198  }
199  }
200  }
201 }

245 int WarnIfPragmaOnceDetected(const std::string& fullHeaderPath,
246  const std::string& headerFileContent)
247 {
248  std::istringstream headerFile(headerFileContent);
249  std::string line;
250  while(std::getline(headerFile,line)){
251  llvm::StringRef lineRef (line);
252  auto trimmedLineRef = lineRef.trim();
253  if (trimmedLineRef.startswith("#pragma") &&
254  (trimmedLineRef.endswith(" once") || trimmedLineRef.endswith("\tonce"))) {
255  std::cerr << "Error: #pragma once directive detected in header file "
256  << fullHeaderPath
257  << " which was requested to be inlined.\n";
258  return 1;
259  }
260  }
261  return 0;
262 }

344 void TModuleGenerator::WriteRegistrationSource(std::ostream &out,
345  const std::string &fwdDeclnArgsToKeepString,
346  const std::string &headersClassesMapString,
347  const std::string &fwdDeclString,
348  const std::string &extraIncludes) const
349 {
350 
351  std::string fwdDeclStringSanitized = fwdDeclString;
352 #ifdef R__WIN32
353  // Visual sudio has a limitation of 2048 characters max in raw strings, so split
354  // the potentially huge DICTFWDDCLS raw string into multiple smaller ones
355  constexpr char from[] = "\n";
356  constexpr char to[] = "\n)DICTFWDDCLS\"\nR\"DICTFWDDCLS(";
357  size_t start_pos = 0;
358  while ((start_pos = fwdDeclStringSanitized.find(from, start_pos)) != std::string::npos) {
359  if (fwdDeclStringSanitized.find(from, start_pos + 1) == std::string::npos) // skip the last
360  break;
361  if ((fwdDeclStringSanitized.at(start_pos + 1) == '}') || (fwdDeclStringSanitized.at(start_pos + 1) == '\n'))
362  start_pos += 2;
363  else {
364  fwdDeclStringSanitized.replace(start_pos, strlen(from), to);
365  start_pos += strlen(to); // In case 'to' contains 'from', like replacing 'x' with 'yx'
366  }
367  }
368 #endif
369  std::string fwdDeclStringRAW;
370  if ("nullptr" == fwdDeclStringSanitized || "\"\"" == fwdDeclStringSanitized) {
371  fwdDeclStringRAW = fwdDeclStringSanitized;
372  } else {
373  fwdDeclStringRAW = "R\"DICTFWDDCLS(\n";
374  fwdDeclStringRAW += "#line 1 \"";
375  fwdDeclStringRAW += fDictionaryName +" dictionary forward declarations' payload\"\n";
376  fwdDeclStringRAW += fwdDeclStringSanitized;
377  fwdDeclStringRAW += ")DICTFWDDCLS\"";
378  }
379 
380  if (fIsInPCH)
381  fwdDeclStringRAW = "nullptr";
382 
383  std::string payloadCode;
384 
385  // Increase the value of the diagnostics pointing out from which
386  // dictionary this payload comes from in case of errors
387  payloadCode += "#line 1 \"";
388  payloadCode += fDictionaryName +" dictionary payload\"\n";
389 
390  // Add defines and undefines to the payloadCode
391  std::ostringstream definesAndUndefines;
392  // Anticipate the undefines.
393  // Suppose to have a namespace called "declarations" used in R5 for template
394  // instantiations in the header given to genreflex.
395  // Now, in this namespace, objects with some names, typically dummy, will be
396  // present.
397  // If you give such headers to cling to parse, problems will occour, as the
398  // names appear multiple times. One possible solution is to get out of this
399  // with preprocessor defines given to genreflex, redefining "declarations"
400  // to a hash or <project>_<package> via the build system.
401  WritePPUndefines(definesAndUndefines);
402  WritePPDefines(definesAndUndefines);
403  payloadCode += definesAndUndefines.str();
404 
405  // If necessary, inline the headers
406  std::string hdrFullPath;
407  std::string inlinedHeaders;
408 
409  auto findAndAddToInlineHeaders = [&](const std::string& hdrName) {
410  bool headerFound = FindHeader(hdrName,hdrFullPath);
411  if (!headerFound) {
412  ROOT::TMetaUtils::Error(0, "Cannot find header %s: cannot inline it.\n", hdrName.c_str());
413  } else {
414  std::ifstream headerFile(hdrFullPath.c_str());
415  const std::string headerFileAsStr((std::istreambuf_iterator<char>(headerFile)),
416  std::istreambuf_iterator<char>());
417  inlinedHeaders += headerFileAsStr;
418  }
419  };
420 
421  if (fInlineInputHeaders) {
422  for (auto const & hdrName : fHeaders) {
423  findAndAddToInlineHeaders(hdrName);
424  }
425 
426  } else {
427  // Now, if not, just #include them in the payload except for the linkdef
428  // file, if any.
429  for (auto & hdrName : fHeaders) {
430  inlinedHeaders += "#include \"" + hdrName + "\"\n";
431  }
432  }
433 
434  if (0 != fLinkDefFile.size() && !fIsPCH) {
435  findAndAddToInlineHeaders(fLinkDefFile);
436  }
437 
438  // Recover old genreflex behaviour, i.e. do not print warnings due to glitches
439  // in the headers at runtime. This is not synonym of ignoring warnings as they
440  // will be printed at dictionary generation time.
441  // In order to do this we leverage the diagnostic pragmas and, since there is no
442  // way to express as a pragma the option "-Wno-deprecated" the
443  // _BACKWARD_BACKWARD_WARNING_H macro, used to avoid to go through
444  // backward/backward_warning.h.
445  payloadCode += "#define _BACKWARD_BACKWARD_WARNING_H\n"
446  "// Inline headers\n"+
447  inlinedHeaders + "\n"+
448  (extraIncludes.empty() ? "" : "// Extra includes\n" + extraIncludes + "\n") +
449  "#undef _BACKWARD_BACKWARD_WARNING_H\n";
450 
451  // Dictionary initialization code for loading the module
452  out << "namespace {\n"
453  " void TriggerDictionaryInitialization_"
454  << GetDictionaryName() << "_Impl() {\n"
455  " static const char* headers[] = {\n";
456  if (fInlineInputHeaders) {
457  out << 0 ;
458  } else {
459  WriteHeaderArray(out);
460  };
461 
462  std::string payloadcodeWrapped = "nullptr";
463  if (!fIsInPCH)
464  payloadcodeWrapped = "R\"DICTPAYLOAD(\n" + payloadCode + ")DICTPAYLOAD\"";
465 
466  out << " };\n"
467  << " static const char* includePaths[] = {\n";
468  WriteIncludePathArray(out) <<
469  " };\n"
470  " static const char* fwdDeclCode = " << fwdDeclStringRAW << ";\n"
471  " static const char* payloadCode = " << payloadcodeWrapped << ";\n"
472  " " << headersClassesMapString << "\n"
473  " static bool isInitialized = false;\n"
474  " if (!isInitialized) {\n"
475  " TROOT::RegisterModule(\"" << GetDemangledDictionaryName() << "\",\n"
476  " headers, includePaths, payloadCode, fwdDeclCode,\n"
477  " TriggerDictionaryInitialization_" << GetDictionaryName() << "_Impl, " << fwdDeclnArgsToKeepString << ", classesHeaders, " << (fCI->getLangOpts().Modules ? "/*has C++ module*/true" : "/*has no C++ module*/false") << ");\n"
478  " isInitialized = true;\n"
479  " }\n"
480  " }\n"
481  " static struct DictInit {\n"
482  " DictInit() {\n"
483  " TriggerDictionaryInitialization_" << GetDictionaryName() << "_Impl();\n"
484  " }\n"
485  " } __TheDictionaryInitializer;\n"
486  "}\n"
487  "void TriggerDictionaryInitialization_" << GetDictionaryName() << "() {\n"
488  " TriggerDictionaryInitialization_" << GetDictionaryName() << "_Impl();\n"
489  "}" << std::endl;
490 }
