182  TString title = Form("%s of %s: Projection %s:%s;%s;%s",
183  cell_value_description.Data(),

185  foam->GetVariableName(i)->String().Data(),
186  foam->GetVariableName(k)->String().Data(),
187  foam->GetVariableName(i)->String().Data(),
