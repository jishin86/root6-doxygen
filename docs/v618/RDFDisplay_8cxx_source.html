105 void RDisplay::AddCollectionToRow(const std::vector<std::string> &collection)
106 {
107  auto row = fCurrentRow;
108  // For each element of the collection, save it. The first element will be in the current row, next ones will have
109  // their own row.
110  size_t collectionSize = collection.size();
111  for (size_t index = 0; index < collectionSize; ++index) {
112  auto stringEle = collection[index];
113  auto element = DElement_t(stringEle);
114 
115  // Update the width if this element is the biggest found
116  if (fWidths[fCurrentColumn] < stringEle.length()) {
117  fWidths[fCurrentColumn] = stringEle.length();
118  }
119 
120  if (index == 0 || index == collectionSize - 1) {
121  // Do nothing, by default DisplayElement is printed
122  } else if (index == 1) {
123  element.SetDots();
124  // Be sure the "..." fit
125  if (fWidths[fCurrentColumn] < 3) {
126  fWidths[fCurrentColumn] = 3;
127  }
128  } else {
129  // In the Print(), after the dots, all element will just be ignored except the last one.
130  element.SetIgnore();
131  }
132 
133  // Save the element
134  fTable[row][fCurrentColumn] = element;
135  ++row;
136 
137  if (index != collectionSize - 1 && fTable.size() <= row) {
138  // This is not the last element, prepare the next row for the next element, if not already done by another
139  // collection
140  fTable.push_back(std::vector<DElement_t>(fNColumns));
141  }
142  }
143  fNextRow = (fNextRow > row) ? fNextRow : row;
144  MovePosition();
145 }

147 void RDisplay::MovePosition()
148 {
149  // Go to the next element. If it is outside the row, just go the first element of the next row.
150  ++fCurrentColumn;
151  if (fCurrentColumn == fNColumns) {
152  fCurrentRow = fNextRow;
153  fCurrentColumn = 0;
154  fNextRow = fCurrentRow + 1;
155  fTable.push_back(std::vector<DElement_t>(fNColumns));
156  }

186 void RDisplay::Print() const
187 {
188  auto columnsToPrint =
189  fNColumns - GetNColumnsToShorten(); // Get the number of columns that fit in the characters limit
190  std::vector<bool> hasPrintedNext(fNColumns,
191  false); // Keeps track if the collection as already been shortened, allowing to skip
192  // all elements until the next printable element.
193  auto nrRows = fTable.size();
194  for (size_t rowIndex = 0; rowIndex < nrRows; ++rowIndex) {
195  auto &row = fTable[rowIndex];
196 
197  std::stringstream stringRow;
198  bool isRowEmpty = true; // It may happen during compacting that some rows are empty, this happens for example if
199  // collections have different size. Thanks to this flag, these rows are just ignored.
200  for (size_t columnIndex = 0; columnIndex < columnsToPrint; ++columnIndex) {
201  const auto &element = row[columnIndex];
202  std::string printedElement = "";
203 
204  if (element.IsDot()) {
205  printedElement = "...";
206  } else if (element.IsPrint()) {
207  // Maybe the element is part of a collection that is being shortened, and so it was already printed.
208  if (!hasPrintedNext[columnIndex]) {
209  printedElement = element.GetRepresentation();
210  }
211  hasPrintedNext[columnIndex] =
212  false; // Encountered "next printable element", shortening can start again when needed.
213  } else { // IsIgnore
214  // Shortening is starting here. Print directly the last element, to have something like 1 ... 3, and don't
215  // print anything else.
216  if (!hasPrintedNext[columnIndex]) {
217  size_t i = rowIndex + 1; // Starting from the next row...
218  for (; !fTable[i][columnIndex].IsPrint(); ++i) {
219  // .. look for the first element that can be printed, it will be the last of the collection.
220  }
221  printedElement = fTable[i][columnIndex].GetRepresentation(); // Print the element
222  hasPrintedNext[columnIndex] = true; // And start ignoring anything else until the next collection.
223  }
224  }
225  if (!printedElement.empty()) {
226  // Found at least one element, so the row is not empty.
227  isRowEmpty = false;
228  }
229 
230  stringRow << std::left << std::setw(fWidths[columnIndex]) << std::setfill(fgSeparator) << printedElement
231  << " | ";
232  }
233  if (!isRowEmpty) {
234  std::cout << stringRow.str() << std::endl;
235  }
236  }
237 }
