203  inline char* DemangleName(const char* mangled_name, int& errorCode)
204  {
205  // Demangle in a portable way the name.
206  // IMPORTANT: The caller is responsible for freeing the returned const char*
207 
208  errorCode=0;
209 #ifdef R__WIN32
210  char *demangled_name = __unDName(0, mangled_name, 0, malloc, free, UNDNAME_COMPLETE);
211  if (!demangled_name) {
212  errorCode = -1;
213  return nullptr;
214  }
215 #else
216  char *demangled_name = abi::__cxa_demangle(mangled_name, 0, 0, &errorCode);
217  if (!demangled_name || errorCode) {
218  free(demangled_name);
219  return nullptr;
220  }
221 #endif
222  return demangled_name;
223  }
