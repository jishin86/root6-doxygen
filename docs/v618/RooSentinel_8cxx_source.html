17 /**
18 \file RooSentinel.cxx
19 \class RooSentinel
20 \ingroup Roofitcore
21 
22 RooSentinel is a special purposes singleton class that terminates
23 all other RooFit singleton services when the process exists.
24 
25 All function RooFit singleton services are created on the heap with
26 a static wrapper function to avoid the 'static initialization order fiasco'
27 but are not automatically destroyed at the end of the session. This class
28 installs an atexit() function that takes care of this
29 **/

70 /// Install atexit handler that calls CleanupRooFitAtExit()
71 /// on program termination
72 
73 void RooSentinel::activate()
74 {
75  if (!_active) {
76  _active = kTRUE ;
77  atexit(CleanUpRooFitAtExit) ;
78  }
79 }
