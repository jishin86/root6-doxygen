349 ////////////////////////////////////////////////////////////////////////////////
350 /// System event loop.
351 
352 void TSystem::Run()
353 {
354  fInControl = kTRUE;
355  fDone = kFALSE;
356 
357 loop_entry:
358  try {
359  RETRY {
360  while (!fDone) {
361  gApplication->StartIdleing();
362  InnerLoop();
363  gApplication->StopIdleing();
364  }
365  } ENDTRY;
366  }
367  catch (std::exception& exc) {
368  TIter next(fStdExceptionHandler);
369  TStdExceptionHandler* eh = 0;
370  while ((eh = (TStdExceptionHandler*) next())) {
371  switch (eh->Handle(exc))
372  {
373  case TStdExceptionHandler::kSEProceed:
374  break;
375  case TStdExceptionHandler::kSEHandled:
376  goto loop_entry;
377  break;
378  case TStdExceptionHandler::kSEAbort:
379  Warning("Run", "instructed to abort");
380  goto loop_end;
381  break;
382  }
383  }
384  throw;
385  }
386  catch (const char *str) {
387  printf("%s\n", str);
388  }
389  // handle every exception
390  catch (...) {
391  Warning("Run", "handle uncaugth exception, terminating");
392  }
393 
394 loop_end:
395  fInControl = kFALSE;
396 }

499 ////////////////////////////////////////////////////////////////////////////////
500 /// Time when next timer of mode (synchronous=kTRUE or
501 /// asynchronous=kFALSE) will time-out (in ms).
502 
503 Long_t TSystem::NextTimeOut(Bool_t mode)
504 {
505  if (!fTimers) return -1;
506 
507  TOrdCollectionIter it((TOrdCollection*)fTimers);
508  TTimer *t, *to = 0;
509  Long64_t tt, tnow = Now();
510  Long_t timeout = -1;
511 
512  while ((t = (TTimer *) it.Next())) {
513  if (t->IsSync() == mode) {
514  tt = (Long64_t)t->GetAbsTime() - tnow;
515  if (tt < 0) tt = 0;
516  if (timeout == -1) {
517  timeout = (Long_t)tt;
518  to = t;
519  }
520  if (tt < timeout) {
521  timeout = (Long_t)tt;
522  to = t;
523  }
524  }
525  }
526 
527  if (to && to->IsAsync() && timeout > 0) {
528  if (to->IsInterruptingSyscalls())
529  SigAlarmInterruptsSyscalls(kTRUE);
530  else
531  SigAlarmInterruptsSyscalls(kFALSE);
532  }
533 
534  return timeout;
535 }

749 ////////////////////////////////////////////////////////////////////////////////
750 /// Create helper TSystem to handle file and directory operations that
751 /// might be special for remote file access.
752 
753 TSystem *TSystem::FindHelper(const char *path, void *dirptr)
754 {
755  if (!fHelpers)
756  fHelpers = new TOrdCollection;
757 
758  TPluginHandler *h;
759  TSystem *helper = 0;
760  if (path) {
761  if (!GetDirPtr()) {
762  TUrl url(path, kTRUE);
763  if (!strcmp(url.GetProtocol(), "file"))
764  return 0;
765  }
766  }
767 
768  // look for existing helpers
769  TIter next(fHelpers);
770  while ((helper = (TSystem*) next()))
771  if (helper->ConsistentWith(path, dirptr))
772  return helper;
773 
774  if (!path)
775  return 0;
776 
777  // create new helper
778  TRegexp re("^root.*:"); // also roots, rootk, etc
779  TString pname = path;
780  if (pname.BeginsWith("xroot:") || pname.Index(re) != kNPOS) {
781  // (x)rootd daemon ...
782  if ((h = gROOT->GetPluginManager()->FindHandler("TSystem", path))) {
783  if (h->LoadPlugin() == -1)
784  return 0;
785  helper = (TSystem*) h->ExecPlugin(2, path, kFALSE);
786  }
787  } else if ((h = gROOT->GetPluginManager()->FindHandler("TSystem", path))) {
788  if (h->LoadPlugin() == -1)
789  return 0;
790  helper = (TSystem*) h->ExecPlugin(0);
791  }
792 
793  if (helper)
794  fHelpers->Add(helper);
795 
796  return helper;
797 }

1001 ////////////////////////////////////////////////////////////////////////////////
1002 /// Return the directory name in pathname. DirName of /user/root is /user.
1003 /// In case no dirname is specified "." is returned.
1004 
1005 const char *TSystem::DirName(const char *pathname)
1006 {
1007  if (pathname && strchr(pathname, '/')) {
1008  R__LOCKGUARD2(gSystemMutex);
1009 
1010  static int len = 0;
1011  static char *buf = 0;
1012  int pathlen = strlen(pathname);
1013  if (pathlen > len) {
1014  delete [] buf;
1015  len = pathlen;
1016  buf = new char [len+1];
1017  }
1018  strcpy(buf, pathname);
1019 
1020  char *r = buf+pathlen-1;
1021  // First skip the trailing '/'
1022  while ( r>buf && *(r)=='/') { --r; }
1023  // Then find the next non slash
1024  while ( r>buf && *(r)!='/') { --r; }
1025  // Then skip duplicate slashes
1026  // Note the 'r>buf' is a strict comparison to allows '/topdir' to return '/'
1027  while ( r>buf && *(r)=='/') { --r; }
1028  // If all was cut away, we encountered a rel. path like 'subdir/'
1029  // and ended up at '.'.
1030  if (r==buf && *(r)!='/') {
1031  return ".";
1032  }
1033  // And finally terminate the string to drop off the filename
1034  *(r+1) = '\0';
1035 
1036  return buf;
1037  }
1038  return ".";
1039 }

1825 ////////////////////////////////////////////////////////////////////////////////
1826 /// Load a shared library. Returns 0 on successful loading, 1 in
1827 /// case lib was already loaded, -1 in case lib does not exist
1828 /// or in case of error and -2 in case of version mismatch.
1829 /// When entry is specified the loaded lib is
1830 /// searched for this entry point (return -1 when entry does not exist,
1831 /// 0 otherwise). When the system flag is kTRUE, the library is considered
1832 /// a permanent system library that should not be unloaded during the
1833 /// course of the session.
1834 
1835 int TSystem::Load(const char *module, const char *entry, Bool_t system)
1836 {
1837  // don't load libraries that have already been loaded
1838  TString libs( GetLibraries() );
1839  TString moduleBasename( BaseName(module) );
1840  TString l(moduleBasename);
1841 
1842  Ssiz_t idx = l.Last('.');
1843  if (idx != kNPOS) {
1844  l.Remove(idx+1);
1845  }
1846  for (idx = libs.Index(l); idx != kNPOS; idx = libs.Index(l,idx+1)) {
1847  // The libs contains the sub-string 'l', let's make sure it is
1848  // not just part of a larger name.
1849  if (idx == 0 || libs[idx-1] == '/' || libs[idx-1] == '\\') {
1850  Ssiz_t len = libs.Length();
1851  idx += l.Length();
1852  if (!l.EndsWith(".") && libs[idx]=='.')
1853  idx++;
1854  // Skip the soversion.
1855  while (idx < len && isdigit(libs[idx])) {
1856  ++idx;
1857  // No need to test for len here, at worse idx==len and lib[idx]=='\0'
1858  if (libs[idx] == '.') {
1859  ++idx;
1860  }
1861  }
1862  while (idx < len && libs[idx] != '.') {
1863  if (libs[idx] == ' ' || idx+1 == len) {
1864  return 1;
1865  }
1866  ++idx;
1867  }
1868  }
1869  }
1870  if (l[l.Length()-1] == '.') {
1871  l.Remove(l.Length()-1);
1872  }
1873  if (l.BeginsWith("lib")) {
1874  l.Replace(0, 3, "-l");
1875  for(idx = libs.Index(l); idx != kNPOS; idx = libs.Index(l,idx+1)) {
1876  if ((idx == 0 || libs[idx-1] == ' ') &&
1877  (libs[idx+l.Length()] == ' ' || libs[idx+l.Length()] == 0)) {
1878  return 1;
1879  }
1880  }
1881  }
1882 
1883  char *path = DynamicPathName(module);
1884 
1885  int ret = -1;
1886  if (path) {
1887  // load any dependent libraries
1888  TString deplibs = gInterpreter->GetSharedLibDeps(moduleBasename);
1889  if (deplibs.IsNull()) {
1890  TString libmapfilename;
1891  libmapfilename = path;
1892  idx = libmapfilename.Last('.');
1893  if (idx != kNPOS) {
1894  libmapfilename.Remove(idx);
1895  }
1896  libmapfilename += ".rootmap";
1897  if (gSystem->GetPathInfo(libmapfilename, 0, (Long_t*)0, 0, 0) == 0) {
1898  if (gDebug > 0) Info("Load", "loading %s", libmapfilename.Data());
1899  gInterpreter->LoadLibraryMap(libmapfilename);
1900  deplibs = gInterpreter->GetSharedLibDeps(moduleBasename);
1901  }
1902  } else {
1903  TString delim(" ");
1904  TObjArray *tokens = deplibs.Tokenize(delim);
1905  for (Int_t i = tokens->GetEntriesFast()-1; i > 0; i--) {
1906  const char *deplib = ((TObjString*)tokens->At(i))->GetName();
1907  if (strcmp(module,deplib)==0) {
1908  continue;
1909  }
1910  if (gDebug > 0)
1911  Info("Load", "loading dependent library %s for library %s",
1912  deplib, ((TObjString*)tokens->At(0))->GetName());
1913  if ((ret = Load(deplib, "", system)) < 0) {
1914  delete tokens;
1915  delete [] path;
1916  return ret;
1917  }
1918  }
1919  delete tokens;
1920  }
1921  if (!system) {
1922  // Mark the library in $ROOTSYS/lib as system.
1923  const char *dirname = DirName(path);
1924  system = R__MatchFilename(TROOT::GetLibDir(), dirname);
1925 
1926  if (!system) {
1927  system = R__MatchFilename(TROOT::GetBinDir(), dirname);
1928  }
1929  }
1930 
1931  gLibraryVersionIdx++;
1932  if (gLibraryVersionIdx == gLibraryVersionMax) {
1933  gLibraryVersionMax *= 2;
1934  gLibraryVersion = TStorage::ReAllocInt(gLibraryVersion, gLibraryVersionMax, gLibraryVersionIdx);
1935  }
1936  ret = gInterpreter->Load(path, system);
1937  if (ret < 0) ret = -1;
1938  if (gDebug > 0)
1939  Info("Load", "loaded library %s, status %d", path, ret);
1940  if (ret == 0 && gLibraryVersion[gLibraryVersionIdx]) {
1941  int v = TROOT::ConvertVersionCode2Int(gLibraryVersion[gLibraryVersionIdx]);
1942  Error("Load", "version mismatch, %s = %d, ROOT = %d",
1943  path, v, gROOT->GetVersionInt());
1944  ret = -2;
1945  gLibraryVersion[gLibraryVersionIdx] = 0;
1946  }
1947  gLibraryVersionIdx--;
1948  delete [] path;
1949  }
1950 
1951  if (!entry || !entry[0] || ret < 0) return ret;
1952 
1953  Func_t f = DynFindSymbol(module, entry);
1954  if (f) return 0;
1955  return -1;
1956 }

1960 /// Returns the number of top level libraries successfully loaded.
1961

1968

2000  }

2745 ////////////////////////////////////////////////////////////////////////////////
2746 /// This method compiles and loads a shared library containing
2747 /// the code from the file "filename".
2748 ///
2749 /// The return value is true (1) in case of success and false (0)
2750 /// in case of error.
2751 ///
2752 /// The possible options are:
2753 /// - k : keep the shared library after the session end.
2754 /// - f : force recompilation.
2755 /// - g : compile with debug symbol
2756 /// - O : optimized the code
2757 /// - c : compile only, do not attempt to load the library.
2758 /// - s : silence all informational output
2759 /// - v : output all information output
2760 /// - d : debug ACLiC, keep all the output files.
2761 /// - - : if buildir is set, use a flat structure (see buildir below)
2762 ///
2763 /// If library_specified is specified, CompileMacro generates the file
2764 /// "library_specified".soext where soext is the shared library extension for
2765 /// the current platform.
2766 ///
2767 /// If build_dir is specified, it is used as an alternative 'root' for the
2768 /// generation of the shared library. The library is stored in a sub-directories
2769 /// of 'build_dir' including the full pathname of the script unless a flat
2770 /// directory structure is requested ('-' option). With the '-' option the libraries
2771 /// are created directly in the directory 'build_dir'; in particular this means that
2772 /// 2 scripts with the same name in different source directory will over-write each
2773 /// other's library.
2774 /// See also TSystem::SetBuildDir.
2775 ///
2776 /// If dirmode is not zero and we need to create the target directory, the
2777 /// file mode bit will be change to 'dirmode' using chmod.
2778 ///
2779 /// If library_specified is not specified, CompileMacro generate a default name
2780 /// for library by taking the name of the file "filename" but replacing the
2781 /// dot before the extension by an underscore and by adding the shared
2782 /// library extension for the current platform.
2783 /// For example on most platform, hsimple.cxx will generate hsimple_cxx.so
2784 ///
2785 /// It uses the directive fMakeSharedLibs to create a shared library.
2786 /// If loading the shared library fails, it tries to output a list of missing
2787 /// symbols by creating an executable (on some platforms like OSF, this does
2788 /// not HAVE to be an executable) containing the script. It uses the
2789 /// directive fMakeExe to do so.
2790 /// For both directives, before passing them to TSystem::Exec, it expands the
2791 /// variables $SourceFiles, $SharedLib, $LibName, $IncludePath, $LinkedLibs,
2792 /// $DepLibs, $ExeName and $ObjectFiles. See SetMakeSharedLib() for more
2793 /// information on those variables.
2794 ///
2795 /// This method is used to implement the following feature:
2796 ///
2797 /// Synopsis:
2798 ///
2799 /// The purpose of this addition is to allow the user to use an external
2800 /// compiler to create a shared library from its C++ macro (scripts).
2801 /// Currently in order to execute a script, a user has to type at the root
2802 /// prompt
2803 /// ~~~ {.cpp}
2804 /// .X myfunc.C(arg1,arg2)
2805 /// ~~~
2806 /// We allow them to type:
2807 /// ~~~ {.cpp}
2808 /// .X myfunc.C++(arg1,arg2)
2809 /// ~~~
2810 /// or
2811 /// ~~~ {.cpp}
2812 /// .X myfunc.C+(arg1,arg2)
2813 /// ~~~
2814 /// In which case an external compiler will be called to create a shared
2815 /// library. This shared library will then be loaded and the function
2816 /// myfunc will be called with the two arguments. With '++' the shared library
2817 /// is always recompiled. With '+' the shared library is recompiled only
2818 /// if it does not exist yet or the macro file is newer than the shared
2819 /// library.
2820 ///
2821 /// Of course the + and ++ notation is supported in similar way for .x and .L.
2822 ///
2823 /// Through the function TSystem::SetMakeSharedLib(), the user will be able to
2824 /// indicate, with shell commands, how to build a shared library (a good
2825 /// default will be provided). The most common change, namely where to find
2826 /// header files, will be available through the function
2827 /// TSystem::SetIncludePath().
2828 /// A good default will be provided so that a typical user session should be at
2829 /// most:
2830 /// ~~~ {.cpp}
2831 /// root[1] gSystem->SetIncludePath("-I$ROOTSYS/include
2832 /// -I$HOME/mypackage/include");
2833 /// root[2] .x myfunc.C++(10,20);
2834 /// ~~~
2835 /// The user may sometimes try to compile a script before it has loaded all the
2836 /// needed shared libraries. In this case we want to be helpfull and output a
2837 /// list of the unresolved symbols. So if the loading of the created shared
2838 /// library fails, we will try to build a executable that contains the
2839 /// script. The linker should then output a list of missing symbols.
2840 ///
2841 /// To support this we provide a TSystem::SetMakeExe() function, that sets the
2842 /// directive telling how to create an executable. The loader will need
2843 /// to be informed of all the libraries available. The information about
2844 /// the libraries that has been loaded by .L and TSystem::Load() is accesible
2845 /// to the script compiler. However, the information about
2846 /// the libraries that have been selected at link time by the application
2847 /// builder (like the root libraries for root.exe) are not available and need
2848 /// to be explicitly listed in fLinkedLibs (either by default or by a call to
2849 /// TSystem::SetLinkedLibs()).
2850 ///
2851 /// To simplify customization we could also add to the .rootrc support for the
2852 /// variables
2853 /// ~~~ {.cpp}
2854 /// Unix.*.Root.IncludePath: -I$ROOTSYS/include
2855 /// WinNT.*.Root.IncludePath: -I%ROOTSYS%/include
2856 ///
2857 /// Unix.*.Root.LinkedLibs: -L$ROOTSYS/lib -lBase ....
2858 /// WinNT.*.Root.LinkedLibs: %ROOTSYS%/lib/*.lib msvcrt.lib ....
2859 /// ~~~
2860 /// And also support for MakeSharedLibs() and MakeExe().

4197  fname[posArgBegin + 1] = 0;
