33  //generate the data for the graphs
34  TRandom r;
35  Int_t i;
36  for (i=0; i<n; i++) {

38  xvalues2[i] = r.Uniform(3, 8);

40  yvalues1[i] = 3 + 2*xvalues1[i] + xvalues1[i]*xvalues1[i] + r.Gaus();

42  evalues1[i] = 1;
43  evalues2[i] = 10;
