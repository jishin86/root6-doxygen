57 void ROOT::Experimental::Detail::RPagePool::CommitPage(const RPage& page)

68 void ROOT::Experimental::Detail::RPagePool::ReleasePage(const RPage& page)
69 {
70  if (page.IsNull()) return;
71  for (unsigned i = 0; i < fNPages; ++i) {
72  if (fPages[i] == page) {
73  if (--fReferences[i] == 0) {
74  fPages[i] = RPage();
75  }
76  return;
77  }
78  }
79  R__ASSERT(false);
80 }

82 ROOT::Experimental::Detail::RPage ROOT::Experimental::Detail::RPagePool::GetPage(RColumn* column, NTupleSize_t index)
83 {
84  for (unsigned i = 0; i < fNPages; ++i) {
85  if (fReferences[i] == 0) continue;
86  if (fPages[i].GetColumnId() != column->GetColumnIdSource()) continue;
87  if (!fPages[i].Contains(index)) continue;
88  fReferences[i]++;
89  return fPages[i];
90  }
91  RPage newPage = ReservePage(column);
92  column->GetPageSource()->PopulatePage(column->GetHandleSource(), index, &newPage);
93  CommitPage(newPage);
94  return newPage;
95 }
