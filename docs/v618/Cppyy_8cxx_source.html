147 std::string Cppyy::GetName( const std::string& name )
148 {
149  if( name.size() == 0) return name;
150  // need to deal with template paremeters that can have scopes themselves
151  Int_t tpl_open = 0;
152  for ( std::string::size_type pos = name.size() - 1; pos > 0; pos-- ) {
153  std::string::value_type c = name[pos];
154  // count '<' and '>' to be able to skip template contents
155  if ( c == '>' )
156  ++tpl_open;
157  else if ( c == '<' )
158  --tpl_open;
159  // by only checking for "::" the last part (class name) is dropped
160  else if ( tpl_open == 0 && c == ':'&& name[ pos - 1 ] == ':' ) {
161  // found a new scope part
162  return name.substr( pos+1 );
163  }
164  }
165  return name;

299 static CallFunc_t* GetCallFunc( Cppyy::TCppMethod_t method )
300 {
301  auto icf = g_method2callfunc.find( method );
302  if ( icf != g_method2callfunc.end() )
303  return icf->second;
304 
305  CallFunc_t* callf = nullptr;
306  TFunction* func = (TFunction*)method;
307  std::string callString = "";
308 
309 // create, if not cached
310  Cppyy::TCppScope_t scope = declaring_scope( method );
311  const TClassRef& klass = type_from_handle( scope );
312  if ( klass.GetClass() || (func && scope == GLOBAL_HANDLE) ) {
313  ClassInfo_t* gcl = klass.GetClass() ? klass->GetClassInfo() : nullptr;
314  if ( ! gcl )
315  gcl = GetGlobalNamespaceInfo();
316 
317  TCollection* method_args = func->GetListOfMethodArgs();
318  TIter iarg( method_args );
319 
320  TMethodArg* method_arg = 0;
321  while ((method_arg = (TMethodArg*)iarg.Next())) {
322  std::string fullType = method_arg->GetTypeNormalizedName();
323  if ( callString.empty() )
324  callString = fullType;
325  else
326  callString += ", " + fullType;
327  }
328 
329  Long_t offset = 0;
330  callf = gInterpreter->CallFunc_Factory();
331 
332  gInterpreter->CallFunc_SetFuncProto(
333  callf,
334  gcl,
335  func ? func->GetName() : klass->GetName(),
336  callString.c_str(),
337  func ? (func->Property() & kIsConstMethod) : kFALSE,
338  &offset,
339  ROOT::kExactMatch );
340 
341 // CLING WORKAROUND -- The number of arguments is not always correct (e.g. when there
342 // are default parameters, causing the callString to be wrong and
343 // the exact match to fail); or the method may have been inline or
344 // be compiler generated. In all those cases the exact match fails,
345 // whereas the conversion match sometimes works.
346  if ( ! gInterpreter->CallFunc_IsValid( callf ) ) {
347  gInterpreter->CallFunc_SetFuncProto(
348  callf,
349  gcl,
350  func ? func->GetName() : klass->GetName(),
351  callString.c_str(),
352  func ? (func->Property() & kIsConstMethod) : kFALSE,
353  &offset ); // <- no kExactMatch as that will fail
354  }
355 // -- CLING WORKAROUND
356 
357  }
358 
359  if ( !( callf && gInterpreter->CallFunc_IsValid( callf ) ) ) {
360  PyErr_Format( PyExc_RuntimeError, "could not resolve %s::%s(%s)",
361  const_cast<TClassRef&>(klass).GetClassName(),
362  func ? func->GetName() : const_cast<TClassRef&>(klass).GetClassName(),
363  callString.c_str() );
364  if ( callf ) gInterpreter->CallFunc_Delete( callf );
365  return nullptr;
366  }
367 
368  g_method2callfunc[ method ] = callf;
369  return callf;

372 static inline void copy_args( void* args_, void** vargs ) {
373  std::vector<TParameter>& args = *(std::vector<TParameter>*)args_;
374  for ( std::vector<TParameter>::size_type i = 0; i < args.size(); ++i ) {
375  switch ( args[i].fTypeCode ) {
376  case 'l': /* long */
377  vargs[i] = (void*)&args[i].fValue.fLong;
378  break;
379  case 'f': /* double */
380  vargs[i] = (void*)&args[i].fValue.fFloat;
381  break;
382  case 'd': /* double */
383  vargs[i] = (void*)&args[i].fValue.fDouble;
384  break;
385  case 'D': /* long double */
386  vargs[i] = (void*)&args[i].fValue.fLongDouble;
387  break;
388  case 'k': /* long long */
389  case 'K': /* unsigned long long */
390  case 'U': /* unsigned long */
391  case 'p': /* void* */
392  vargs[i] = (void*)&args[i].fValue.fVoidp;
393  break;
394  case 'V': /* (void*)type& */
395  vargs[i] = args[i].fValue.fVoidp;
396  break;
397  case 'r': /* const type& */
398  vargs[i] = args[i].fRef;
399  break;
400  default:
401  std::cerr << "unknown type code: " << args[i].fTypeCode << std::endl;
402  break;
403  }
404  }

407 Bool_t FastCall(
408  Cppyy::TCppMethod_t method, void* args_, void* self, void* result )
409 {
410  const std::vector<TParameter>& args = *(std::vector<TParameter>*)args_;
411 
412  CallFunc_t* callf = GetCallFunc( method );
413  if ( ! callf )
414  return kFALSE;
415 
416  TInterpreter::CallFuncIFacePtr_t faceptr = gCling->CallFunc_IFacePtr( callf );
417  if ( faceptr.fKind == TInterpreter::CallFuncIFacePtr_t::kGeneric ) {
418  if ( args.size() <= SMALL_ARGS_N ) {
419  void* smallbuf[SMALL_ARGS_N];
420  copy_args( args_, smallbuf );
421  faceptr.fGeneric( self, args.size(), smallbuf, result );
422  } else {
423  std::vector<void*> buf( args.size() );
424  copy_args( args_, buf.data() );
425  faceptr.fGeneric( self, args.size(), buf.data(), result );
426  }
427  return kTRUE;
428  }
429 
430  if ( faceptr.fKind == TInterpreter::CallFuncIFacePtr_t::kCtor ) {
431  if ( args.size() <= SMALL_ARGS_N ) {
432  void* smallbuf[SMALL_ARGS_N];
433  copy_args( args_, (void**)smallbuf );
434  faceptr.fCtor( (void**)smallbuf, result, args.size() );
435  } else {
436  std::vector<void*> buf( args.size() );
437  copy_args( args_, buf.data() );
438  faceptr.fCtor( buf.data(), result, args.size() );
439  }
440  return kTRUE;
441  }
442 
443  if ( faceptr.fKind == TInterpreter::CallFuncIFacePtr_t::kDtor ) {
444  std::cerr << " DESTRUCTOR NOT IMPLEMENTED YET! " << std::endl;
445  return kFALSE;
446  }
447 
448  return kFALSE;
449 }
