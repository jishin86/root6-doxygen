36 /// Obtain a reader lock. Returns always 0.

38 Int_t TRWLock::ReadLock()
39 {
40  fMutex.Lock();
41 
42  while (fWriters)
43  fLockFree.Wait();
44 
45  fReaders++;
46 
47  fMutex.UnLock();
48 
49  return 0;
50 }

52 ////////////////////////////////////////////////////////////////////////////////
53 /// Unlock reader lock. Returns -1 if thread was not locked,
54 /// 0 if everything ok.
55 
56 Int_t TRWLock::ReadUnLock()
57 {
58  fMutex.Lock();
59 
60  if (fReaders == 0) {
61  fMutex.UnLock();
62  return -1;
63  } else {
64  fReaders--;
65  if (fReaders == 0)
66  fLockFree.Signal();
67  fMutex.UnLock();
68  return 0;
69  }
70 }
