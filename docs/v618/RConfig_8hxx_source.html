586 /*---- unlikely / likely expressions -----------------------------------------*/
587 // These are meant to use in cases like:
588 // if (R__unlikely(expression)) { ... }
589 // in performance-critical sessions. R__unlikely / R__likely provide hints to
590 // the compiler code generation to heavily optimize one side of a conditional,
591 // causing the other branch to have a heavy performance cost.
592 //
593 // It is best to use this for conditionals that test for rare error cases or
594 // backward compatibility code.
595 
596 #if (__GNUC__ >= 3) || defined(__INTEL_COMPILER)
597 #if !defined(R__unlikely)
598  #define R__unlikely(expr) __builtin_expect(!!(expr), 0)
599 #endif
600 #if !defined(R__likely)
601  #define R__likely(expr) __builtin_expect(!!(expr), 1)
602 #endif
603 #else
604  #define R__unlikely(expr) expr
605  #define R__likely(expr) expr
606 #endif
