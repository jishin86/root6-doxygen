674 
675 void RooVectorDataStore::weightError(Double_t& lo, Double_t& hi, RooAbsData::ErrorType etype) const
676 {
677  if (_extWgtArray) {
678 
679  // We have a weight array, use that info
680  switch (etype) {
681 
682  case RooAbsData::Auto:
683  throw string(Form("RooDataHist::weightError(%s) error type Auto not allowed here",GetName())) ;
684  break ;
685 
686  case RooAbsData::Expected:
687  throw string(Form("RooDataHist::weightError(%s) error type Expected not allowed here",GetName())) ;
688  break ;
689 
690  case RooAbsData::Poisson:
691  // Weight may be preset or precalculated
692  if (_curWgtErrLo>=0) {
693  lo = _curWgtErrLo ;
694  hi = _curWgtErrHi ;
695  return ;
696  }
697 
698  // Otherwise Calculate poisson errors
699  Double_t ym,yp ;
700  RooHistError::instance().getPoissonInterval(Int_t(weight()+0.5),ym,yp,1) ;
701  lo = weight()-ym ;
702  hi = yp-weight() ;
703  return ;
704 
705  case RooAbsData::SumW2:
706  lo = _curWgtErr ;
707  hi = _curWgtErr ;
