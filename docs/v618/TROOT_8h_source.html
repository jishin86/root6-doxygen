16 //////////////////////////////////////////////////////////////////////////
17 // //
18 // TROOT //
19 // //
20 // The TROOT object is the entry point to the system. //
21 // The single instance of TROOT is accessible via the global gROOT. //
22 // Using the gROOT pointer one has access to basically every object //
23 // created in a ROOT based program. The TROOT object is essentially a //
24 // "dispatcher" with several lists pointing to the ROOT main objects. //
25 // //

398  /// \brief call RecursiveRemove for obj if gROOT is valid
399  /// and obj.TestBit(kMustCleanup) is true.
400  /// Note: this reset the kMustCleanup bit to allow
401  /// harmless multiple call to this function.
402  inline void CallRecursiveRemoveIfNeeded(TObject &obj)
403  {
404  if (obj.TestBit(kMustCleanup)) {
405  TROOT *root = ROOT::Internal::gROOTLocal;
406  if (root && root != &obj && (root->MustClean() || Internal::RequiresCleanup(obj))) {
407  root->RecursiveRemove(&obj);
408  obj.ResetBit(kMustCleanup);
409  }
410  }
411  }
