11 void padRefresh(TPad *pad,int flag=0)
12 {
13  if (!pad) return;
14  pad->Modified();
15  pad->Update();
16  TList *tl = pad->GetListOfPrimitives();
17  if (!tl) return;
18  TListIter next(tl);
19  TObject *to;
20  while ((to=next())) {
21  if (to->InheritsFrom(TPad::Class())) padRefresh((TPad*)to,1);}
22  if (flag) return;
23  gSystem->ProcessEvents();
24 }
