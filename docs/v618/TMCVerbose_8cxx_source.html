239 ////////////////////////////////////////////////////////////////////////////////
240 /// Stepping info
241 
242 void TMCVerbose::Stepping()
243 {
244  if (fLevel>2) {
245 
246 #if __GNUC__ >= 3
247  std::cout << std::fixed;
248 #endif
249 
250  // Step number
251  //
252  std::cout << "#" << std::setw(4) << fStepNumber++ << " ";
253 
254  // Position
255  //
256  Double_t x, y, z;
257  gMC->TrackPosition(x, y, z);
258  std::cout << std::setw(8) << std::setprecision(3) << x << " "
259  << std::setw(8) << std::setprecision(3) << y << " "
260  << std::setw(8) << std::setprecision(3) << z << " ";
261 
262  // Kinetic energy
263  //
264  Double_t px, py, pz, etot;
265  gMC->TrackMomentum(px, py, pz, etot);
266  Double_t ekin = etot - gMC->TrackMass();
267  std::cout << std::setw(9) << std::setprecision(4) << ekin*1e03 << " ";
268 
269  // Energy deposit
270  //
271  std::cout << std::setw(9) << std::setprecision(4) << gMC->Edep()*1e03 << " ";
272 
273  // Step length
274  //
275  std::cout << std::setw(8) << std::setprecision(3) << gMC->TrackStep() << " ";
276 
277  // Track length
278  //
279  std::cout << std::setw(8) << std::setprecision(3) << gMC->TrackLength() << " ";
280 
281  // Volume
282  //
283  if (gMC->CurrentVolName() != 0)
284  std::cout << std::setw(4) << gMC->CurrentVolName() << " ";
285  else
286  std::cout << std::setw(4) << "None" << " ";
287 
288  // Process
289  //
290  TArrayI processes;
291  Int_t nofProcesses = gMC->StepProcesses(processes);
292  if (nofProcesses > 0)
293  std::cout << TMCProcessName[processes[nofProcesses-1]];
294 
295  std::cout << std::endl;
296  }
297 }
