27 /// Return the length, if any, taken by std:: and any
28 /// potential inline namespace (well compiler detail namespace).

30 static size_t StdLen(const std::string_view name)
31 {
32  size_t len = 0;
33  if (name.compare(0,5,"std::")==0) {
34  len = 5;
35 
36  // TODO: This is likely to induce unwanted autoparsing, those are reduced
37  // by the caching of the result.
38  if (gInterpreterHelper) {
39  for(size_t i = 5; i < name.length(); ++i) {
40  if (name[i] == '<') break;
41  if (name[i] == ':') {
42  bool isInlined;
43  std::string scope(name.data(),i);
44  std::string scoperesult;
45  // We assume that we are called in already serialized code.
46  // Note: should we also cache the negative answers?
47  static std::set<std::string> gInlined;
48 
49  if (gInlined.find(scope) != gInlined.end()) {
50  len = i;
51  if (i+1<name.length() && name[i+1]==':') {
52  len += 2;
53  }
54  }
55  if (!gInterpreterHelper->ExistingTypeCheck(scope, scoperesult)
56  && gInterpreterHelper->IsDeclaredScope(scope,isInlined)) {
57  if (isInlined) {
58  gInlined.insert(scope);
59  len = i;
60  if (i+1<name.length() && name[i+1]==':') {
61  len += 2;
62  }
63  }
64  }
65  }
66  }
67  }
68  }
69 
70  return len;
71 }

157 {

162

165  // this couldn't possibly be an STL container

1422  // A::B::C< D, E::F, G::H<I,J>::K::L >::M

1424  // and G::H<I,J>::K or G might be a typedef.

1427 
1428  if (tname[cursor]==' ') {
1429  if (!modified) {
1430  modified = true;
1431  result += string(tname,0,cursor);
1432  }
1433  while (tname[cursor]==' ') ++cursor;
1434  }
1435 
1436  if (tname[cursor]=='c' && (cursor+6<len)) {
1437  if (strncmp(tname+cursor,"const ",6) == 0) {
1438  cursor += 6;
1439  if (modified) result += "const ";
1440  }
1441  constprefix = true;
1442 
1443  }
1444 
1445  // When either of those two is true, we should probably go to modified
1446  // mode. (Otherwise we rely on somebody else to strip the std::)
1447  if (len > 5 && strncmp(tname+cursor,"std::",5) == 0) {
1448  cursor += 5;
1449  }
1450  if (len > 2 && strncmp(tname+cursor,"::",2) == 0) {
1451  cursor += 2;
1452  len -= 2;
1453  }
1454 
1455  unsigned int start_of_type = cursor;
1456  unsigned int end_of_type = 0;
1457  unsigned int mod_start_of_type = result.length();

1459  for ( ; cursor<len; ++cursor) {
1460  switch (tname[cursor]) {
1461  case ':': {
1462  if ((cursor+1)>=len || tname[cursor+1]!=':') {
1463  // we expected another ':', malformed, give up.
1464  if (modified) result += (tname+prevScope);
1465  return;
1466  }

1468  if (modified) {
1469  scope = result.substr(mod_start_of_type, string::npos);
1470  scope += std::string(tname+prevScope,cursor-prevScope);
1471  } else {
1472  scope = std::string(tname, start_of_type, cursor - start_of_type); // we need to try to avoid this copy
1473  }
1474  std::string scoperesult;
1475  bool isInlined = false;
1476  if (gInterpreterHelper->ExistingTypeCheck(scope, scoperesult)
1477  ||gInterpreterHelper->GetPartiallyDesugaredNameWithScopeHandling(scope, scoperesult)) {
1478  // it is a known type
1479  if (!scoperesult.empty()) {
1480  // and it is a typedef
1481  if (modified) {
1482  if (constprefix && scoperesult.compare(0,6,"const ",6) != 0) mod_start_of_type -= 6;
1483  result.replace(mod_start_of_type, string::npos,
1484  scoperesult);
1485  result += "::";
1486  } else {
1487  modified = true;
1488  mod_start_of_type = start_of_type;
1489  result += string(tname,0,start_of_type);
1490  //if (constprefix) result += "const ";
1491  result += scoperesult;
1492  result += "::";
1493  }
1494  } else if (modified) {
1495  result += std::string(tname+prevScope,cursor+1-prevScope);
1496  }
1497  } else if (!gInterpreterHelper->IsDeclaredScope(scope,isInlined)) {
1498  // the nesting namespace is not declared
1499  if (modified) result += (tname+prevScope);
1500  // Unfortunately, this is too harsh .. what about:
1501  // unknown::wrapper<Int_t>
1502  return;
1503  } else if (isInlined) {
1504  // humm ... just skip it.
1505  if (!modified) {
1506  modified = true;
1507  mod_start_of_type = start_of_type;
1508  result += string(tname,0,start_of_type);
1509  //if (constprefix) result += "const ";
1510  result += string(tname,start_of_type,prevScope - start_of_type);
1511  }
1512  } else if (modified) {
1513  result += std::string(tname+prevScope,cursor+1-prevScope);
1514  }
1515  // Consume the 1st semi colon, the 2nd will be consume by the for loop.
1516  ++cursor;
1517  prevScope = cursor+1;

1519  }
1520  case '<': {
1521  // push information on stack
1522  if (modified) {
1523  result += std::string(tname+prevScope,cursor+1-prevScope);
1524  // above includes the '<' .... result += '<';
1525  }

1529  } while( cursor<len && tname[cursor] == ',' );
1530 
1531  while (cursor<len && tname[cursor+1]==' ') ++cursor;
1532

1534  // (respective the scope section and final type processing section)
1535  // as they would re-do the same job.
1536  if (cursor+2<len && tname[cursor+1]==':' && tname[cursor+2]==':') {
1537  if (modified) result += "::";
1538  cursor += 2;
1539  prevScope = cursor+1;
1540  }
1541  if ( (cursor+1)<len && tname[cursor+1] == ',') {
1542  ++cursor;
1543  if (modified) result += ',';
1544  return;
1545  }
1546  if ( (cursor+1)<len && tname[cursor+1] == '>') {
1547  ++cursor;
1548  if (modified) result += " >";
1549  return;
1550  }
1551  if ( (cursor+1) >= len) {
1552  return;
1553  }
1554  if (tname[cursor] != ' ') break;
1555  if (modified) prevScope = cursor+1;
1556  // If the 'current' character is a space we need to treat it,
1557  // since this the next case statement, we can just fall through,
1558  // otherwise we should need to do:
1559  // --cursor; break;
1560  }
1561  case ' ': {
1562  end_of_type = cursor;
1563  // let's see if we have 'long long' or 'unsigned int' or 'signed char' or what not.
1564  while ((cursor+1)<len && tname[cursor+1] == ' ') ++cursor;
1565 
1566  auto next = cursor+1;
1567  if (strncmp(tname+next,"const",5) == 0 && ((next+5)==len || tname[next+5] == ' ' || tname[next+5] == '*' || tname[next+5] == '&' || tname[next+5] == ',' || tname[next+5] == '>' || tname[next+5] == ']'))
1568  {
1569  // A first const after the type needs to be move in the front.
1570  if (!modified) {
1571  modified = true;
1572  result += string(tname,0,start_of_type);
1573  result += "const ";
1574  mod_start_of_type = start_of_type + 6;
1575  result += string(tname,start_of_type,end_of_type-start_of_type);
1576  } else if (mod_start_of_type < result.length()) {
1577  result.insert(mod_start_of_type,"const ");
1578  mod_start_of_type += 6;
1579  } else {
1580  result += "const ";
1581  mod_start_of_type += 6;
1582  result += string(tname,start_of_type,end_of_type-start_of_type);
1583  }
1584  cursor += 5;
1585  end_of_type = cursor+1;
1586  prevScope = end_of_type;
1587  if ((next+5)==len || tname[next+5] == ',' || tname[next+5] == '>' || tname[next+5] == '[') {
1588  break;
1589  }
1590  } else if (next!=len && tname[next] != '*' && tname[next] != '&') {
1591  // the type is not ended yet.
1592  end_of_type = 0;
1593  break;
1594  }
1595  ++cursor;
1596  // Intentional fall through;
1597  }
1598  case '*':
1599  case '&': {
1600  if (tname[cursor] != ' ') end_of_type = cursor;
1601  // check and skip const (followed by *,&, ,) ... what about followed by ':','['?
1602  auto next = cursor+1;
1603  if (strncmp(tname+next,"const",5) == 0) {
1604  if ((next+5)==len || tname[next+5] == ' ' || tname[next+5] == '*' || tname[next+5] == '&' || tname[next+5] == ',' || tname[next+5] == '>' || tname[next+5] == '[') {
1605  next += 5;
1606  }
1607  }
1608  while (next<len &&
1609  (tname[next] == ' ' || tname[next] == '*' || tname[next] == '&')) {
1610  ++next;
1611  // check and skip const (followed by *,&, ,) ... what about followed by ':','['?

1613  if ((next+5)==len || tname[next+5] == ' ' || tname[next+5] == '*' || tname[next+5] == '&' || tname[next+5] == ',' || tname[next+5] == '>' || tname[next+5] == '[') {
1614  next += 5;
1615  }
1616  }
1617  }
1618  cursor = next-1;
1619 // if (modified && mod_start_of_type < result.length()) {
1620 // result += string(tname,end_of_type,cursor-end_of_type);
1621 // }
1622  break;
1623  }
1624  case ',': {
1625  if (modified && prevScope) {
1626  result += std::string(tname+prevScope,(end_of_type == 0 ? cursor : end_of_type)-prevScope);
1627  }
1628  ResolveTypedefProcessType(tname,len,cursor,constprefix,start_of_type,end_of_type,mod_start_of_type,
1629  modified, result);
1630  if (modified) result += ',';
1631  return;
1632  }
1633  case '>': {
1634  if (modified && prevScope) {
1635  result += std::string(tname+prevScope,(end_of_type == 0 ? cursor : end_of_type)-prevScope);
1636  }
1637  ResolveTypedefProcessType(tname,len,cursor,constprefix,start_of_type,end_of_type,mod_start_of_type,
1638  modified, result);
1639  if (modified) result += '>';
1640  return;
1641  }
1642  default:
1643  end_of_type = 0;
1644  }
1645  }
1646 
1647  if (prevScope && modified) result += std::string(tname+prevScope,(end_of_type == 0 ? cursor : end_of_type)-prevScope);
1648 
1649  ResolveTypedefProcessType(tname,len,cursor,constprefix,start_of_type,end_of_type,mod_start_of_type,
1650  modified, result);
