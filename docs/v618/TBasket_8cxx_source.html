236 ////////////////////////////////////////////////////////////////////////////////
237 /// Load basket buffers in memory without unziping.
238 /// This function is called by TTreeCloner.
239 /// The function returns 0 in case of success, 1 in case of error.
240 
241 Int_t TBasket::LoadBasketBuffers(Long64_t pos, Int_t len, TFile *file, TTree *tree)
242 {
243  if (fBufferRef) {
244  // Reuse the buffer if it exist.
245  fBufferRef->Reset();
246 
247  // We use this buffer both for reading and writing, we need to
248  // make sure it is properly sized for writing.
249  fBufferRef->SetWriteMode();
250  if (fBufferRef->BufferSize() < len) {
251  fBufferRef->Expand(len);
252  }
253  fBufferRef->SetReadMode();
254  } else {
255  fBufferRef = new TBufferFile(TBuffer::kRead, len);
256  }
257  fBufferRef->SetParent(file);
258  char *buffer = fBufferRef->Buffer();
259  file->Seek(pos);
260  TFileCacheRead *pf = tree->GetReadCache(file);
261  if (pf) {
262  TVirtualPerfStats* temp = gPerfStats;
263  if (tree->GetPerfStats()) gPerfStats = tree->GetPerfStats();
264  Int_t st = pf->ReadBuffer(buffer,pos,len);
265  if (st < 0) {
266  return 1;
267  } else if (st == 0) {
268  // fOffset might have been changed via TFileCacheRead::ReadBuffer(), reset it
269  file->Seek(pos);
270  // If we are using a TTreeCache, disable reading from the default cache
271  // temporarily, to force reading directly from file
272  TTreeCache *fc = dynamic_cast<TTreeCache*>(file->GetCacheRead());
273  if (fc) fc->Disable();
274  Int_t ret = file->ReadBuffer(buffer,len);
275  if (fc) fc->Enable();
276  pf->AddNoCacheBytesRead(len);
277  pf->AddNoCacheReadCalls(1);
278  if (ret) {
279  return 1;
280  }
281  }
282  gPerfStats = temp;
283  // fOffset might have been changed via TFileCacheRead::ReadBuffer(), reset it
284  file->SetOffset(pos + len);
285  } else {
286  TVirtualPerfStats* temp = gPerfStats;
287  if (tree->GetPerfStats() != 0) gPerfStats = tree->GetPerfStats();
288  if (file->ReadBuffer(buffer,len)) {
289  gPerfStats = temp;
290  return 1; //error while reading
291  }
292  else gPerfStats = temp;
293  }
294 
295  fBufferRef->SetReadMode();
296  fBufferRef->SetBufferOffset(0);
297  Streamer(*fBufferRef);
298 
299  return 0;
300 }

303 /// Remove the first dentries of this basket, moving entries at
304 /// dentries to the start of the buffer.
305 
306 void TBasket::MoveEntries(Int_t dentries)
307 {
308  Int_t i;
309 
310  if (dentries >= fNevBuf) return;
311  Int_t bufbegin;
312  Int_t moved;
313 
314  Int_t *entryOffset = GetEntryOffset();
315  if (entryOffset) {
316  bufbegin = entryOffset[dentries];
317  moved = bufbegin-GetKeylen();
318 
319  // First store the original location in the fDisplacement array
320  // and record the new start offset
321 
322  if (!fDisplacement) {
323  fDisplacement = new Int_t[fNevBufSize];
324  }
325  for (i = 0; i<(fNevBufSize-dentries); ++i) {
326  fDisplacement[i] = entryOffset[i + dentries];
327  entryOffset[i] = entryOffset[i + dentries] - moved;
328  }
329  for (i = fNevBufSize-dentries; i<fNevBufSize; ++i) {
330  fDisplacement[i] = 0;
331  entryOffset[i] = 0;
332  }
333 
334  } else {
335  // If there is no EntryOffset array, this means
336  // that each entry has the same size and that
337  // it does not point to other objects (hence there
338  // is no need for a displacement array).
339  bufbegin = GetKeylen() + dentries*fNevBufSize;
340  moved = bufbegin-GetKeylen();
341  }
342  TBuffer *buf = GetBufferRef();
343  char *buffer = buf->Buffer();
344  memmove(buffer+GetKeylen(),buffer+bufbegin,buf->Length()-bufbegin);
345  buf->SetBufferOffset(buf->Length()-moved);
346  fNevBuf -= dentries;
347 }

445 ////////////////////////////////////////////////////////////////////////////////
446 /// Read basket buffers in memory and cleanup.
447 ///
448 /// Read a basket buffer. Check if buffers of previous ReadBasket
449 /// should not be dropped. Remember, we keep buffers in memory up to
450 /// fMaxVirtualSize.
451 /// The function returns 0 in case of success, 1 in case of error
452 /// This function was modified with the addition of the parallel
453 /// unzipping, it will try to get the unzipped file from the cache
454 /// receiving only a pointer to that buffer (so we shall not
455 /// delete that pointer), although we get a new buffer in case
456 /// it's not found in the cache.
457 /// There is a lot of code duplication but it was necesary to assure
458 /// the expected behavior when there is no cache.
459 
460 Int_t TBasket::ReadBasketBuffers(Long64_t pos, Int_t len, TFile *file)
461 {
462  if(!fBranch->GetDirectory()) {
463  return -1;
464  }
465 
466  Bool_t oldCase;
467  char *rawUncompressedBuffer, *rawCompressedBuffer;
468  Int_t uncompressedBufferLen;
469 
470  // See if the cache has already unzipped the buffer for us.
471  TFileCacheRead *pf = nullptr;
472  {
473  R__LOCKGUARD_IMT(gROOTMutex); // Lock for parallel TTree I/O
474  pf = fBranch->GetTree()->GetReadCache(file);
475  }
476  if (pf) {
477  Int_t res = -1;
478  Bool_t free = kTRUE;
479  char *buffer = nullptr;
480  res = pf->GetUnzipBuffer(&buffer, pos, len, &free);
481  if (R__unlikely(res >= 0)) {
482  len = ReadBasketBuffersUnzip(buffer, res, free, file);
483  // Note that in the kNotDecompressed case, the above function will return 0;
484  // In such a case, we should stop processing
485  if (len <= 0) return -len;
486  goto AfterBuffer;
487  }
488  }
489 
490  // Determine which buffer to use, so that we can avoid a memcpy in case of
491  // the basket was not compressed.
492  TBuffer* readBufferRef;
493  if (R__unlikely(fBranch->GetCompressionLevel()==0)) {
494  readBufferRef = fBufferRef;
495  } else {
496  readBufferRef = fCompressedBufferRef;
497  }
498 
499  // fBufferSize is likely to be change in the Streamer call (below)
500  // and we will re-add the new size later on.
501  fBranch->GetTree()->IncrementTotalBuffers(-fBufferSize);
502 
503  // Initialize the buffer to hold the compressed data.
504  readBufferRef = R__InitializeReadBasketBuffer(readBufferRef, len, file);
505  if (!readBufferRef) {
506  Error("ReadBasketBuffers", "Unable to allocate buffer.");
507  return 1;
508  }
509 
510  if (pf) {
511  TVirtualPerfStats* temp = gPerfStats;
512  if (fBranch->GetTree()->GetPerfStats() != 0) gPerfStats = fBranch->GetTree()->GetPerfStats();
513  Int_t st = 0;
514  {
515  R__LOCKGUARD_IMT(gROOTMutex); // Lock for parallel TTree I/O
516  st = pf->ReadBuffer(readBufferRef->Buffer(),pos,len);
517  }
518  if (st < 0) {
519  return 1;
520  } else if (st == 0) {
521  // Read directly from file, not from the cache
522  // If we are using a TTreeCache, disable reading from the default cache
523  // temporarily, to force reading directly from file
524  R__LOCKGUARD_IMT(gROOTMutex); // Lock for parallel TTree I/O
525  TTreeCache *fc = dynamic_cast<TTreeCache*>(file->GetCacheRead());
526  if (fc) fc->Disable();
527  Int_t ret = file->ReadBuffer(readBufferRef->Buffer(),pos,len);
528  if (fc) fc->Enable();
529  pf->AddNoCacheBytesRead(len);
530  pf->AddNoCacheReadCalls(1);
531  if (ret) {
532  return 1;
533  }
534  }
535  gPerfStats = temp;
536  } else {
537  // Read from the file and unstream the header information.
538  TVirtualPerfStats* temp = gPerfStats;
539  if (fBranch->GetTree()->GetPerfStats() != 0) gPerfStats = fBranch->GetTree()->GetPerfStats();
540  R__LOCKGUARD_IMT(gROOTMutex); // Lock for parallel TTree I/O
541  if (file->ReadBuffer(readBufferRef->Buffer(),pos,len)) {
542  gPerfStats = temp;
543  return 1;
544  }
545  else gPerfStats = temp;
546  }
547  Streamer(*readBufferRef);
548  if (IsZombie()) {
549  return 1;
550  }
551 
552  rawCompressedBuffer = readBufferRef->Buffer();
553 
554  // Are we done?
555  if (R__unlikely(readBufferRef == fBufferRef)) // We expect most basket to be compressed.
556  {
557  if (R__likely(fObjlen+fKeylen == fNbytes)) {
558  // The basket was really not compressed as expected.
559  goto AfterBuffer;
560  } else {
561  // Well, somehow the buffer was compressed anyway, we have the compressed data in the uncompressed buffer
562  // Make sure the compressed buffer is initialized, and memcpy.
563  InitializeCompressedBuffer(len, file);
564  if (!fCompressedBufferRef) {
565  Error("ReadBasketBuffers", "Unable to allocate buffer.");
566  return 1;
567  }
568  fBufferRef->Reset();
569  rawCompressedBuffer = fCompressedBufferRef->Buffer();
570  memcpy(rawCompressedBuffer, fBufferRef->Buffer(), len);
571  }
572  }
573 
574  // Initialize buffer to hold the uncompressed data
575  // Note that in previous versions we didn't allocate buffers until we verified
576  // the zip headers; this is no longer beforehand as the buffer lifetime is scoped
577  // to the TBranch.
578  uncompressedBufferLen = len > fObjlen+fKeylen ? len : fObjlen+fKeylen;
579  fBufferRef = R__InitializeReadBasketBuffer(fBufferRef, uncompressedBufferLen, file);
580  rawUncompressedBuffer = fBufferRef->Buffer();
581  fBuffer = rawUncompressedBuffer;
582 
583  oldCase = OLD_CASE_EXPRESSION;
584  // Case where ROOT thinks the buffer is compressed. Copy over the key and uncompress the object
585  if (fObjlen > fNbytes-fKeylen || oldCase) {
586  if (R__unlikely(TestBit(TBufferFile::kNotDecompressed) && (fNevBuf==1))) {
587  return ReadBasketBuffersUncompressedCase();
588  }
589 
590  // Optional monitor for zip time profiling.
591  Double_t start = 0;
592  if (R__unlikely(gPerfStats)) {
593  start = TTimeStamp();
594  }
595 
596  memcpy(rawUncompressedBuffer, rawCompressedBuffer, fKeylen);
597  char *rawUncompressedObjectBuffer = rawUncompressedBuffer+fKeylen;
598  UChar_t *rawCompressedObjectBuffer = (UChar_t*)rawCompressedBuffer+fKeylen;
599  Int_t nin, nbuf;
600  Int_t nout = 0, noutot = 0, nintot = 0;
601 
602  // Unzip all the compressed objects in the compressed object buffer.
603  while (1) {
604  // Check the header for errors.
605  if (R__unlikely(R__unzip_header(&nin, rawCompressedObjectBuffer, &nbuf) != 0)) {
606  Error("ReadBasketBuffers", "Inconsistency found in header (nin=%d, nbuf=%d)", nin, nbuf);
607  break;
608  }
609  if (R__unlikely(oldCase && (nin > fObjlen || nbuf > fObjlen))) {
610  //buffer was very likely not compressed in an old version
611  memcpy(rawUncompressedBuffer+fKeylen, rawCompressedObjectBuffer+fKeylen, fObjlen);
612  goto AfterBuffer;
613  }
614 
615  R__unzip(&nin, rawCompressedObjectBuffer, &nbuf, (unsigned char*) rawUncompressedObjectBuffer, &nout);
616  if (!nout) break;
617  noutot += nout;
618  nintot += nin;
619  if (noutot >= fObjlen) break;
620  rawCompressedObjectBuffer += nin;
621  rawUncompressedObjectBuffer += nout;
622  }
623 
624  // Make sure the uncompressed numbers are consistent with header.
625  if (R__unlikely(noutot != fObjlen)) {
626  Error("ReadBasketBuffers", "fNbytes = %d, fKeylen = %d, fObjlen = %d, noutot = %d, nout=%d, nin=%d, nbuf=%d", fNbytes,fKeylen,fObjlen, noutot,nout,nin,nbuf);
627  fBranch->GetTree()->IncrementTotalBuffers(fBufferSize);
628  return 1;
629  }
630  len = fObjlen+fKeylen;
631  TVirtualPerfStats* temp = gPerfStats;
632  if (fBranch->GetTree()->GetPerfStats() != 0) gPerfStats = fBranch->GetTree()->GetPerfStats();
633  if (R__unlikely(gPerfStats)) {
634  gPerfStats->UnzipEvent(fBranch->GetTree(),pos,start,nintot,fObjlen);
635  }
636  gPerfStats = temp;
637  } else {
638  // Nothing is compressed - copy over wholesale.
639  memcpy(rawUncompressedBuffer, rawCompressedBuffer, len);
640  }
641 
642 AfterBuffer:
643 
644  fBranch->GetTree()->IncrementTotalBuffers(fBufferSize);
645 
646  // Read offsets table if needed.
647  // If there's no EntryOffsetLen in the branch -- or the fEntryOffset is marked to be calculated-on-demand --
648  // then we skip reading out.
649  if (!fBranch->GetEntryOffsetLen() || (fEntryOffset == reinterpret_cast<Int_t *>(-1))) {
650  return 0;
651  }
652  // At this point, we're required to read out an offset array.
653  ResetEntryOffset(); // TODO: every basket, we reset the offset array. Is this necessary?
654  // Could we instead switch to std::vector?

723 /// Reset the basket to the starting state. i.e. as it was after calling
724 /// the constructor (and potentially attaching a TBuffer.)

726 
727 void TBasket::Reset()
728 {

730  fResetAllocation = false;
731 #ifdef R__TRACK_BASKET_ALLOC_TIME
732  fResetAllocationTime = 0;
733 #endif
734 
735  // Name, Title, fClassName, fBranch
736  // stay the same.
737 
738  // Downsize the buffer if needed.
739  // See if our current buffer size is significantly larger (>2x) than the historical average.
740  // If so, try decreasing it at this flush boundary to closer to the size from OptimizeBaskets
741  // (or this historical average).
742  Int_t curSize = fBufferRef->BufferSize();
743  // fBufferLen at this point is already reset, so use indirect measurements
744  Int_t curLen = (GetObjlen() + GetKeylen());
745  Long_t newSize = -1;
746  if (curSize > 2*curLen)
747  {
748  Long_t curBsize = fBranch->GetBasketSize();
749  if (curSize > 2*curBsize ) {
750  Long_t avgSize = (Long_t)(fBranch->GetTotBytes() / (1+fBranch->GetWriteBasket())); // Average number of bytes per basket so far
751  if (curSize > 2*avgSize) {
752  newSize = curBsize;
753  if (curLen > newSize) {
754  newSize = curLen;
755  }
756  if (avgSize > newSize) {
757  newSize = avgSize;
758  }
759  newSize = newSize + 512 - newSize%512; // Wiggle room and alignment (512 is same as in OptimizeBaskets)
760  }
761  }
762  }
763  // If fBufferRef grew since we last saw it, shrink it to "target memory ratio" of the occupied size
764  // This discourages us from having poorly-occupied buffers on branches with little variability.
765  //
766  // Does not help protect against a burst in event sizes, but does help in the cases where the basket
767  // size jumps from 4MB to 8MB while filling the basket, but we only end up utilizing 4.1MB.
768  //
769  // The above code block is meant to protect against extremely large events.
770 
771  Float_t target_mem_ratio = fBranch->GetTree()->GetTargetMemoryRatio();
772  Int_t max_size = TMath::Max(fLastWriteBufferSize[0], std::max(fLastWriteBufferSize[1], fLastWriteBufferSize[2]));
773  Int_t target_size = static_cast<Int_t>(target_mem_ratio * Float_t(max_size));
774  if (max_size && (curSize > target_size) && (newSize == -1)) {
775  newSize = target_size;
776  newSize = newSize + 512 - newSize % 512; // Wiggle room and alignment, as above.
777  // We only bother with a resize if it saves 8KB (two normal memory pages).
778  if ((newSize > curSize - 8 * 1024) ||
779  (static_cast<Float_t>(curSize) / static_cast<Float_t>(newSize) < target_mem_ratio)) {
780  newSize = -1;
781  } else if (gDebug > 0) {
782  Info("Reset", "Resizing to %ld bytes (was %d); last three sizes were [%d, %d, %d].", newSize, curSize,
783  fLastWriteBufferSize[0], fLastWriteBufferSize[1], fLastWriteBufferSize[2]);
784  }
785  }
786

788  fResetAllocation = true;
789 #ifdef R__TRACK_BASKET_ALLOC_TIME
790  std::chrono::time_point<std::chrono::system_clock> start, end;
791  start = std::chrono::high_resolution_clock::now();
792 #endif
793  fBufferRef->Expand(newSize,kFALSE); // Expand without copying the existing data.
794 #ifdef R__TRACK_BASKET_ALLOC_TIME
795  end = std::chrono::high_resolution_clock::now();

797  fResetAllocationTime = us.count();
798 #endif
799  }

802  fLastWriteBufferSize[fNextBufferSizeRecord] = curLen;
803  fNextBufferSizeRecord = (fNextBufferSizeRecord + 1) % 3;

806

808  if (newNevBufSize==0) {
809  ResetEntryOffset();
810  } else if ((newNevBufSize != fNevBufSize) || (fEntryOffset <= reinterpret_cast<Int_t *>(-1))) {
811  ResetEntryOffset();
812  fEntryOffset = new Int_t[newNevBufSize];
813  }
814  fNevBufSize = newNevBufSize;
815 
816  fNevBuf = 0;
817  Int_t *storeEntryOffset = fEntryOffset;
818  fEntryOffset = 0;
819  Int_t *storeDisplacement = fDisplacement;
820  fDisplacement= 0;
821  fBuffer = 0;
822 
823  fBufferRef->Reset();
824  fBufferRef->SetWriteMode();
825 
826  fHeaderOnly = kTRUE;
827  fLast = 0; //Must initialize before calling Streamer()
828 
829  Streamer(*fBufferRef);
830 
831  fKeylen = fBufferRef->Length();
832  fObjlen = fBufferSize - fKeylen;
833  fLast = fKeylen;
834  fBuffer = 0;
835  fHeaderOnly = kFALSE;
836  fDisplacement= storeDisplacement;
837  fEntryOffset = storeEntryOffset;
838  if (fNevBufSize) {
839  for (Int_t i=0;i<fNevBufSize;i++) fEntryOffset[i] = 0;
840  }

862 /// Stream a class object.
863 
864 void TBasket::Streamer(TBuffer &b)
865 {
866  // As in TBranch::GetBasket, this is used as a half-hearted measure to suppress
867  // the error reporting when many failures occur.
868  static std::atomic<Int_t> nerrors(0);
869 
870  char flag;
871  if (b.IsReading()) {
872  TKey::Streamer(b); //this must be first
873  Version_t v = b.ReadVersion();
874  b >> fBufferSize;
875  // NOTE: we now use the upper-bit of the fNevBufSize to see if we have serialized any of the
876  // optional IOBits. If that bit is set, we immediately read out the IOBits; to replace this
877  // (minimal) safeguard against corruption, we will set aside the upper-bit of fIOBits to do
878  // the same thing (the fact this bit is reserved is tested in the unit tests). If there is
879  // someday a need for more than 7 IOBits, we'll widen the field using the same trick.
880  //
881  // We like to keep this safeguard because we immediately will allocate a buffer based on
882  // the value of fNevBufSize -- and would like to avoid wildly inappropriate allocations.
883  b >> fNevBufSize;
884  if (fNevBufSize < 0) {
885  fNevBufSize = -fNevBufSize;
886  b >> fIOBits;
887  if (!fIOBits || (fIOBits & (1 << 7))) {
888  Error("TBasket::Streamer",
889  "The value of fNevBufSize (%d) or fIOBits (%d) is incorrect ; setting the buffer to a zombie.",
890  -fNevBufSize, fIOBits);
891  MakeZombie();
892  fNevBufSize = 0;
893  } else if (fIOBits && (fIOBits & ~static_cast<Int_t>(EIOBits::kSupported))) {
894  nerrors++;
895  if (nerrors < 10) {
896  Error("Streamer", "The value of fIOBits (%s) contains unknown flags (supported flags "
897  "are %s), indicating this was written with a newer version of ROOT "
898  "utilizing critical IO features this version of ROOT does not support."
899  " Refusing to deserialize.",
900  std::bitset<32>(static_cast<Int_t>(fIOBits)).to_string().c_str(),
901  std::bitset<32>(static_cast<Int_t>(EIOBits::kSupported)).to_string().c_str());
902  } else if (nerrors == 10) {
903  Error("Streamer", "Maximum number of errors has been reported; disabling further messages"
904  "from this location until the process exits.");
905  }
906  fNevBufSize = 0;
907  MakeZombie();
908  }
909  }
910  b >> fNevBuf;
911  b >> fLast;
912  b >> flag;
913  if (fLast > fBufferSize) fBufferSize = fLast;
914  Bool_t mustGenerateOffsets = false;
915  if (flag >= 80) {
916  mustGenerateOffsets = true;
917  flag -= 80;
918  }
919  if (!mustGenerateOffsets && flag && (flag % 10 != 2)) {
920  ResetEntryOffset();
921  fEntryOffset = new Int_t[fNevBufSize];
922  if (fNevBuf) b.ReadArray(fEntryOffset);
923  if (20<flag && flag<40) {
924  for(int i=0; i<fNevBuf; i++){
925  fEntryOffset[i] &= ~kDisplacementMask;
926  }
927  }
928  if (flag>40) {
929  fDisplacement = new Int_t[fNevBufSize];
930  b.ReadArray(fDisplacement);
931  }
932  } else if (mustGenerateOffsets) {
933  // We currently believe that in all cases when offsets can be generated, then the
934  // displacement array must be zero.
935  assert(flag <= 40);
936  fEntryOffset = reinterpret_cast<Int_t *>(-1);
937  }
938  if (flag == 1 || flag > 10) {
939  fBufferRef = new TBufferFile(TBuffer::kRead,fBufferSize);
940  fBufferRef->SetParent(b.GetParent());
941  char *buf = fBufferRef->Buffer();
942  if (v > 1) b.ReadFastArray(buf,fLast);
943  else b.ReadArray(buf);
944  fBufferRef->SetBufferOffset(fLast);
945  // This is now done in the TBranch streamer since fBranch might not
946  // yet be set correctly.
947  // fBranch->GetTree()->IncrementTotalBuffers(fBufferSize);
948  }
949  } else {
950 
951  TKey::Streamer(b); //this must be first
952  b.WriteVersion(TBasket::IsA());
953  if (fBufferRef) {
954  Int_t curLast = fBufferRef->Length();
955  if (!fHeaderOnly && !fSeekKey && curLast > fLast) fLast = curLast;
956  }
957  if (fLast > fBufferSize) fBufferSize = fLast;
958 
959  b << fBufferSize;
960  if (fIOBits) {
961  b << -fNevBufSize;
962  b << fIOBits;
963  } else {
964  b << fNevBufSize;
965  }
966  b << fNevBuf;
967  b << fLast;
968  Bool_t mustGenerateOffsets = fEntryOffset && fNevBuf &&
969  (fIOBits & static_cast<UChar_t>(TBasket::EIOBits::kGenerateOffsetMap)) &&
970  CanGenerateOffsetArray();
971  // We currently believe that in all cases when offsets can be generated, then the
972  // displacement array must be zero.
973  assert(!mustGenerateOffsets || fDisplacement == nullptr);
974  if (fHeaderOnly) {
975  flag = mustGenerateOffsets ? 80 : 0;
976  b << flag;
977  } else {
978  // On return from this function, we are guaranteed that fEntryOffset
979  // is either a valid pointer or nullptr.
980  if (fNevBuf) {
981  GetEntryOffset();
982  }
983  flag = 1;
984  if (!fNevBuf || !fEntryOffset)
985  flag = 2;
986  if (fBufferRef) flag += 10;
987  if (fDisplacement) flag += 40;
988  // Test if we can skip writing out the offset map.
989  if (mustGenerateOffsets) {
990  flag += 80;
991  }
992  b << flag;
993 
994  if (!mustGenerateOffsets && fEntryOffset && fNevBuf) {
995  b.WriteArray(fEntryOffset, fNevBuf);
996  if (fDisplacement) b.WriteArray(fDisplacement, fNevBuf);
997  }
998  if (fBufferRef) {
999  char *buf = fBufferRef->Buffer();
1000  b.WriteFastArray(buf, fLast);
1001  }
1002  }
1003  }
