27 /** \class TMCManager
28  \ingroup vmc
29 
30 Singleton manager class for handling and steering a run with multiple TVirtualMC
31 engines sharing events.
32 It provides interfaces to transfer tracks between engines and exposes the correct
33 stacks to each running engine. A registered user stack is kept up-to-date
34 automatically seeing a consistent history.
35 Track objects (aka TParticle) are still owned by the user who must forward these to
36 the manager after creation. Everything else is done automatically.
37 */

80 ///
81 /// A TVirtualMC will register itself via this method during construction
82 /// if a TMCManager was instanciated before.
83 /// The TMCManager will assign an ID to the engines.
84 ///
85 
86 void TMCManager::Register(TVirtualMC *mc)
87 {
88  // Do not register an engine twice.
89  for (auto &currMC : fEngines) {
90  if (currMC == mc) {
91  ::Fatal("TMCManager::RegisterMC", "This engine is already registered.");
92  }
93  }
94  // Set id and register.
95  mc->SetId(fEngines.size());
96  fEngines.push_back(mc);
97  fStacks.emplace_back(new TMCManagerStack());
98  mc->SetStack(fStacks.back().get());
99  mc->SetManagerStack(fStacks.back().get());
100  // Don't attempt to call TVirtualMCApplication hooks related to geometry
101  // construction
102  mc->SetExternalGeometryConstruction();
103  mc->SetExternalParticleGeneration();
104  // Must update engine pointers here since during construction of the concrete TVirtualMC
105  // implementation the static TVirtualMC::GetMC() or defined gMC might be used.
106  UpdateEnginePointers(mc);
107 }

110 ///
111 /// The user application will register itself via this method when the
112 /// manager was requested.
113 ///
114 
115 void TMCManager::Register(TVirtualMCApplication *application)
116 {
117  if (fApplication) {
118  ::Fatal("TMCManager::Register", "The application is already registered.");
119  }
120  ::Info("TMCManager::Register", "Register user application and construct geometry");
121  fApplication = application;
122  // TODO Can these 3 functions can be called directly here? Or could any of these depend on an implemented VMC?
123  fApplication->ConstructGeometry();
124  fApplication->MisalignGeometry();
125  fApplication->ConstructOpGeometry();
126  if (!gGeoManager->IsClosed()) {
127  // Setting the top volume is the duty of the user as well as closing it.
128  // Failing here is just an additional cross check. If not closed the user
129  // might have forgotten something.
130  ::Fatal("TMCManager::Register", "The TGeo geometry is not closed. Please check whether you just have to close "
131  "it or whether something was forgotten.");
132  }
133 }

231 ////////////////////////////////////////////////////////////////////////////////
232 ///
233 /// User interface to forward particle to specifiic engine.
234 /// It is assumed that the TParticle is owned by the user. It will not be
235 /// modified by the TMCManager.
236 ///
237 
238 void TMCManager::ForwardTrack(Int_t toBeDone, Int_t trackId, Int_t parentId, TParticle *particle, Int_t engineId)
239 {
240  if (engineId < 0 || engineId >= static_cast<Int_t>(fEngines.size())) {
241  ::Fatal("TMCManager::ForwardTrack", "Engine ID %i out of bounds. Have %zu engines.", engineId, fEngines.size());
242  }
243  if (trackId >= static_cast<Int_t>(fParticles.size())) {
244  fParticles.resize(trackId + 1, nullptr);
245  fParticlesStatus.resize(trackId + 1);
246  }
247  fParticles[trackId] = particle;
248  fParticlesStatus[trackId].reset(new TMCParticleStatus());
249  fParticlesStatus[trackId]->fId = trackId;
250  fParticlesStatus[trackId]->fParentId = parentId;
251  fParticlesStatus[trackId]->InitFromParticle(particle);
252  fTotalNTracks++;
253  if (particle->IsPrimary()) {
254  fTotalNPrimaries++;
255  }
256 
257  if (toBeDone > 0) {
258  if (particle->IsPrimary()) {
259  fStacks[engineId]->PushPrimaryTrackId(trackId);
260  } else {
261  fStacks[engineId]->PushSecondaryTrackId(trackId);
262  }
263  }
264 }

294 ////////////////////////////////////////////////////////////////////////////////
295 ///
296 /// Transfer track from current engine to target engine mc
297 ///
298 
299 void TMCManager::TransferTrack(TVirtualMC *mc)
300 {
301  // Do nothing if target and current engines are the same
302  if (mc == fCurrentEngine) {
303  return;
304  }
305 
306  // Get information on current track and extract status from transporting engine
307  Int_t trackId = fStacks[fCurrentEngine->GetId()]->GetCurrentTrackNumber();
308 
309  fCurrentEngine->TrackPosition(fParticlesStatus[trackId]->fPosition);
310  fCurrentEngine->TrackMomentum(fParticlesStatus[trackId]->fMomentum);
311  fCurrentEngine->TrackPolarization(fParticlesStatus[trackId]->fPolarization);
312  fParticlesStatus[trackId]->fStepNumber = fCurrentEngine->StepNumber();
313  fParticlesStatus[trackId]->fTrackLength = fCurrentEngine->TrackLength();
314  fParticlesStatus[trackId]->fWeight = fCurrentEngine->TrackWeight();
315 
316  TGeoBranchArray *geoState = fBranchArrayContainer.GetNewGeoState(fParticlesStatus[trackId]->fGeoStateIndex);
317  geoState->InitFromNavigator(gGeoManager->GetCurrentNavigator());
318 
319  // Push only the particle ID
320  if (fParticles[trackId]->IsPrimary()) {
321  fStacks[mc->GetId()]->PushPrimaryTrackId(trackId);
322  } else {
323  fStacks[mc->GetId()]->PushSecondaryTrackId(trackId);
324  }
325  fCurrentEngine->InterruptTrack();
326 }

328 ////////////////////////////////////////////////////////////////////////////////
329 ///
330 /// Initialize engines
331 ///
332 
333 void TMCManager::Init()
334 {
335  if (fIsInitialized) {
336  return;
337  }
338 
339  if (!fUserStack) {
340  ::Fatal("TMCManager::Run", "Missing user stack pointer.");
341  }
342  if (fEngines.empty()) {
343  ::Fatal("TMCManager::Run", "No engines registered");
344  }
345 
346  for (auto &mc : fEngines) {
347  // Must have geometry handling via TGeo
348  if (!mc->IsRootGeometrySupported()) {
349  ::Fatal("TMCManager::Run", "Engine %s does not support geometry built via ROOT's TGeoManager", mc->GetName());
350  }
351  Int_t currentEngineId = mc->GetId();
352  // To be able to forward info to the user stack
353  fStacks[currentEngineId]->SetUserStack(fUserStack);
354  // Connect the engine's stack to the centrally managed vectors
355  fStacks[currentEngineId]->ConnectTrackContainers(&fParticles, &fParticlesStatus, &fBranchArrayContainer,
356  &fTotalNPrimaries, &fTotalNTracks);
357  }
358 
359  // Initialize the fBranchArrayContainer to manage and cache TGeoBranchArrays
360  fBranchArrayContainer.InitializeFromGeoManager(gGeoManager);
361 
362  fIsInitialized = kTRUE;
363 
364  // Send warning if only one engine ==> overhead
365  if (fEngines.size() == 1) {
366  ::Warning("TMCManager::Run", "Only one engine registered. That will lead to overhead in "
367  "the simulation run due to additional hooks and dispatches "
368  "to the TMCManager.");
369  }
370 }

372 ////////////////////////////////////////////////////////////////////////////////
373 ///
374 /// Run the event loop
375 ///
376 
377 void TMCManager::Run(Int_t nEvents)
378 {
379  if (!fIsInitialized) {
380  ::Fatal("TMCManager::Run", "Engines have not yet been initialized.");
381  }
382 
383  // Set user initialize true in any case now so that this cannot happen by accident during a run
384  fIsInitializedUser = kTRUE;
385 
386  if (nEvents < 1) {
387  ::Fatal("TMCManager::Run", "Need at least one event to process but %i events specified.", nEvents);
388  }
389 
390  // Run 1 event nEvents times
391  for (Int_t i = 0; i < nEvents; i++) {
392  ::Info("TMCManager::Run", "Start event %i", i + 1);
393  PrepareNewEvent();
394  fApplication->BeginEvent();
395  // Loop as long as there are tracks in any engine stack
396  while (GetNextEngine()) {
397  fCurrentEngine->ProcessEvent(i, kTRUE);
398  }
399  fApplication->FinishEvent();
400  }
401  TerminateRun();
402 }
403 
404 ////////////////////////////////////////////////////////////////////////////////
405 ///
406 /// Choose next engines to be run in the loop
407 ///
408 
409 void TMCManager::PrepareNewEvent()
410 {
411  fBranchArrayContainer.FreeGeoStates();
412  // Reset in event flag for all engines and clear stacks
413  for (auto &stack : fStacks) {
414  stack->ResetInternals();
415  }
416  for (UInt_t i = 0; i < fParticles.size(); i++) {
417  fParticlesStatus.clear();
418  fParticlesStatus.resize(fParticles.size());
419  fParticles[i] = nullptr;
420  }
421 
422  // GeneratePrimaries centrally
423  fApplication->GeneratePrimaries();
424 }
425 
426 ////////////////////////////////////////////////////////////////////////////////
427 ///
428 /// Choose next engines to be run in the loop
429 ///
430 
431 Bool_t TMCManager::GetNextEngine()
432 {
433  // Select next engine based on finite number of particles on the stack
434  for (UInt_t i = 0; i < fStacks.size(); i++) {
435  if (fStacks[i]->GetStackedNtrack() > 0) {
436  UpdateEnginePointers(fEngines[i]);
437  return kTRUE;
438  }
439  }
440  // No tracks to be processed.
441  return kFALSE;
442 }
