25 template<typename Real_t>
26 Real_t TReference<Real_t>::L1Regularization(const TMatrixT<Real_t> & W)
27 {
28  size_t m,n;
29  m = W.GetNrows();
30  n = W.GetNcols();
31 
32  Real_t result = 0.0;
33 
34  for (size_t i = 0; i < m; i++) {
35  for (size_t j = 0; j < n; j++) {
36  result += std::abs(W(i,j));
37  }
38  }
39  return result;
40 }
