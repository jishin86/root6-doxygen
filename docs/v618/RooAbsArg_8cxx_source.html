18 /** \class RooAbsArg
19  \ingroup Roofitcore
20 
21 RooAbsArg is the common abstract base class for objects that
22 represent a value (of arbitrary type) and "shape" that in general
23 depends on (is a client of) other RooAbsArg subclasses. The only
24 state information about a value that is maintained in this base
25 class consists of named attributes and flags that track when either
26 the value or the shape of this object changes. The meaning of shape
27 depends on the client implementation but could be, for example, the
28 allowed range of a value. The base class is also responsible for
29 managing client/server links and propagating value/shape changes
30 through an expression tree. RooAbsArg implements public interfaces
31 for inspecting client/server relationships and
32 setting/clearing/testing named attributes.
33 
34 */

868 /// Substitute our servers with those listed in newSet. If nameChange is false, servers and
869 /// and substitutes are matched by name. If nameChange is true, servers are matched to args
870 /// in newSet that have the 'ORIGNAME:<servername>' attribute set. If mustReplaceAll is set,
871 /// a warning is printed and error status is returned if not all servers could be successfully
872 /// substituted.
873 
874 Bool_t RooAbsArg::redirectServers(const RooAbsCollection& newSetOrig, Bool_t mustReplaceAll, Bool_t nameChange, Bool_t isRecursionStep)
875 {
876  // Trivial case, no servers
877  if (_serverList.empty()) return kFALSE ;
878  if (newSetOrig.getSize()==0) return kFALSE ;
879 
880  // Strip any non-matching removal nodes from newSetOrig
881  RooAbsCollection* newSet ;
882 
883  if (nameChange) {
884  newSet = new RooArgSet ;
885  for (auto arg : newSetOrig) {
886 
887  if (string("REMOVAL_DUMMY")==arg->GetName()) {
888 
889  if (arg->getAttribute("REMOVE_ALL")) {
890 // cout << "RooAbsArg::redir including remove_all node " << arg->GetName() << endl ;
891  newSet->add(*arg) ;
892  } else if (arg->getAttribute(Form("REMOVE_FROM_%s",getStringAttribute("ORIGNAME")))) {
893 // cout << "RooAbsArg::redir including remove_from_" << GetName() << " node " << arg->GetName() << endl ;
894  newSet->add(*arg) ;
895  }
896  } else {
897  newSet->add(*arg) ;
898  }
899  }
900 
901 // cout << "RooAbsArg::redirect with name change(" << GetName() << ") newSet = " << newSet << " origSet = " << newSetOrig << endl ;
902 
903  } else {
904  newSet = (RooAbsCollection*) &newSetOrig ;
905  }
906 
907  // Replace current servers with new servers with the same name from the given list
908  Bool_t ret(kFALSE) ;
909 
910  //Copy original server list to not confuse the iterator while deleting
911  std::vector<RooAbsArg*> origServerList, origServerValue, origServerShape;
912  auto origSize = _serverList.size();
913  origServerList.reserve(origSize);
914  origServerValue.reserve(origSize);
915 
916  for (const auto oldServer : _serverList) {
917  origServerList.push_back(oldServer) ;
918 
919  // Retrieve server side link state information
920  if (oldServer->_clientListValue.containsByNamePtr(this)) {
921  origServerValue.push_back(oldServer) ;
922  }
923  if (oldServer->_clientListShape.containsByNamePtr(this)) {
924  origServerShape.push_back(oldServer) ;
925  }
926  }
927 
928  // Delete all previously registered servers
929  for (auto oldServer : origServerList) {
930 
931  RooAbsArg * newServer= oldServer->findNewServer(*newSet, nameChange);
932 
933  if (newServer && _verboseDirty) {
934  cxcoutD(LinkStateMgmt) << "RooAbsArg::redirectServers(" << (void*)this << "," << GetName() << "): server " << oldServer->GetName()
935  << " redirected from " << oldServer << " to " << newServer << endl ;
936  }
937 
938  if (!newServer) {
939  if (mustReplaceAll) {
940  coutE(LinkStateMgmt) << "RooAbsArg::redirectServers(" << (void*)this << "," << GetName() << "): server " << oldServer->GetName()
941  << " (" << (void*)oldServer << ") not redirected" << (nameChange?"[nameChange]":"") << endl ;
942  ret = kTRUE ;
943  }
944  continue ;
945  }
946 
947  auto findByNamePtr = [&oldServer](const RooAbsArg * item) {
948  return oldServer->namePtr() == item->namePtr();
949  };
950  bool propValue = std::any_of(origServerValue.begin(), origServerValue.end(), findByNamePtr);
951  bool propShape = std::any_of(origServerShape.begin(), origServerShape.end(), findByNamePtr);
952  // cout << "replaceServer with name " << oldServer->GetName() << " old=" << oldServer << " new=" << newServer << endl ;
953  if (newServer != this) {
954  replaceServer(*oldServer,*newServer,propValue,propShape) ;
955  }
956  }
957 
958 
959  setValueDirty() ;
960  setShapeDirty() ;
961 
962  // Process the proxies
963  Bool_t allReplaced=kTRUE ;
964  for (int i=0 ; i<numProxies() ; i++) {
965  RooAbsProxy* p = getProxy(i) ;
966  if (!p) continue ;
967  Bool_t ret2 = p->changePointer(*newSet,nameChange,kFALSE) ;
968  allReplaced &= ret2 ;
969  }
970 
971  if (mustReplaceAll && !allReplaced) {
972  coutE(LinkStateMgmt) << "RooAbsArg::redirectServers(" << GetName()
973  << "): ERROR, some proxies could not be adjusted" << endl ;
974  ret = kTRUE ;
975  }
976 
977  // Optional subclass post-processing
978  for (Int_t i=0 ;i<numCaches() ; i++) {
979  ret |= getCache(i)->redirectServersHook(*newSet,mustReplaceAll,nameChange,isRecursionStep) ;
980  }
981  ret |= redirectServersHook(*newSet,mustReplaceAll,nameChange,isRecursionStep) ;
982 
983  if (nameChange) {
984  delete newSet ;
985  }
986 
987  return ret ;
988 }

1033 Bool_t RooAbsArg::recursiveRedirectServers(const RooAbsCollection& newSet, Bool_t mustReplaceAll, Bool_t nameChange, Bool_t recurseInNewSet)
1034 {
1035  // Recursively redirect all servers with new server in collection 'newSet'.
1036  // Substitute our servers with those listed in newSet. If nameChange is false, servers and
1037  // and substitutes are matched by name. If nameChange is true, servers are matched to args
1038  // in newSet that have the 'ORIGNAME:<servername>' attribute set. If mustReplaceAll is set,
1039  // a warning is printed and error status is returned if not all servers could be sucessfully
1040  // substituted. If recurseInNewSet is true, the recursion algorithm also recursion into
1041  // expression trees under the arguments in the new servers (i.e. those in newset)
1042 
1043 
1044  // Cyclic recursion protection
1045  static std::set<const RooAbsArg*> callStack;
1046  {
1047  std::set<const RooAbsArg*>::iterator it = callStack.lower_bound(this);
1048  if (it != callStack.end() && this == *it) {
1049  return kFALSE;
1050  } else {
1051  callStack.insert(it, this);
1052  }
1053  }
1054 
1055  // Do not recurse into newset if not so specified
1056 // if (!recurseInNewSet && newSet.contains(*this)) {
1057 // return kFALSE ;
1058 // }
1059 
1060 
1061  // Apply the redirectServers function recursively on all branch nodes in this argument tree.
1062  Bool_t ret(kFALSE) ;
1063 
1064  cxcoutD(LinkStateMgmt) << "RooAbsArg::recursiveRedirectServers(" << this << "," << GetName() << ") newSet = " << newSet << " mustReplaceAll = "
1065  << (mustReplaceAll?"T":"F") << " nameChange = " << (nameChange?"T":"F") << " recurseInNewSet = " << (recurseInNewSet?"T":"F") << endl ;
1066 
1067  // Do redirect on self (identify operation as recursion step)
1068  ret |= redirectServers(newSet,mustReplaceAll,nameChange,kTRUE) ;
1069

1071  for (const auto server : _serverList){
1072  ret |= server->recursiveRedirectServers(newSet,mustReplaceAll,nameChange,recurseInNewSet) ;
1073  }
1074 
1075  callStack.erase(this);
1076  return ret ;
1077 }
